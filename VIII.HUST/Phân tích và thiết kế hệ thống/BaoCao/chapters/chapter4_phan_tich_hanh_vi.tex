\chapter{PHÂN TÍCH HÀNH VI}

\section{Sơ đồ lớp tổng thể}

Dựa trên các thực thể nghiệp vụ đã phân tích ở Chương 2 và Chương 3, hệ thống được thiết kế với các lớp chính sau:

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{NguoiDung}: Lưu thông tin tài khoản người dùng, bao gồm tên đăng nhập, mật khẩu (đã mã hóa), họ tên, vai trò và trạng thái. Quản lý xác thực và phân quyền.
    \item \textbf{HangHoa}: Lưu thông tin hàng hóa: mã, tên, nhóm hàng, đơn vị tính, giá nhập, giá xuất, mô tả, trạng thái. Là lớp trung tâm của hệ thống.
    \item \textbf{NhomHangHoa}: Lưu thông tin nhóm hàng hóa để phân loại. Một nhóm hàng có nhiều hàng hóa.
    \item \textbf{DonViTinh}: Lưu đơn vị đo lường hàng hóa (cái, kg, hộp, thùng...). Một đơn vị tính dùng cho nhiều hàng hóa.
    \item \textbf{NhaCungCap}: Lưu thông tin nhà cung cấp: mã, tên, địa chỉ, liên hệ, email. Liên kết với phiếu nhập kho.
    \item \textbf{KhachHang}: Lưu thông tin khách hàng: mã, tên, địa chỉ, liên hệ, email, mã số thuế. Liên kết với phiếu xuất kho và đơn hàng.
    \item \textbf{Kho}: Lưu thông tin kho hàng: mã, tên, địa chỉ, người phụ trách, diện tích. Một hệ thống có nhiều kho.
    \item \textbf{PhieuNhapKho}: Lưu thông tin phiếu nhập: mã, nhà cung cấp, kho, người lập, ngày nhập, tổng tiền, trạng thái.
    \item \textbf{ChiTietPhieuNhap}: Lưu chi tiết hàng hóa trong phiếu nhập: mã phiếu, mã hàng, số lượng, đơn giá, thành tiền.
    \item \textbf{PhieuXuatKho}: Lưu thông tin phiếu xuất: mã, khách hàng, kho, người lập, ngày xuất, tổng tiền, trạng thái.
    \item \textbf{ChiTietPhieuXuat}: Lưu chi tiết hàng hóa trong phiếu xuất: mã phiếu, mã hàng, số lượng, đơn giá, thành tiền.
    \item \textbf{PhieuKiemKe}: Lưu thông tin phiếu kiểm kê: mã, kho, người kiểm, ngày kiểm, trạng thái.
    \item \textbf{ChiTietKiemKe}: Lưu chi tiết kiểm kê: mã phiếu, mã hàng, số lượng thực tế, số lượng sổ sách, chênh lệch.
    \item \textbf{TonKho}: Lưu số lượng tồn kho của từng hàng hóa tại từng kho. Được cập nhật khi nhập/xuất/kiểm kê.
    \item \textbf{DonDatHang}: Lưu thông tin đơn đặt hàng: mã, khách hàng, ngày đặt, ngày giao, trạng thái.
    \item \textbf{ChiTietDonHang}: Lưu chi tiết đơn hàng: mã đơn, mã hàng, số lượng, giá, thành tiền.
    \item \textbf{ThanhToan}: Lưu thông tin thanh toán: mã, phiếu liên quan, số tiền, ngày, phương thức.
    \item \textbf{LichSuThaoTac}: Lưu log thao tác: mã, người dùng, hành động, thời gian, đối tượng.
\end{itemize}

Sơ đồ lớp tổng thể được trình bày thành ba sơ đồ con để dễ quan sát, tránh chồng chéo và đảm bảo chữ hiển thị rõ ràng:

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Sơ đồ lớp danh mục (Master Data)}: bao gồm các lớp lưu thông tin cơ bản, ít thay đổi của hệ thống: \texttt{NguoiDung}, \texttt{HangHoa}, \texttt{NhomHangHoa}, \texttt{DonViTinh}, \texttt{NhaCungCap}, \texttt{KhachHang}, \texttt{Kho}, \texttt{TonKho}.
    \item \textbf{Sơ đồ lớp nghiệp vụ nhập xuất}: bao gồm các lớp liên quan đến phiếu nhập kho và phiếu xuất kho: \texttt{PhieuNhapKho}, \texttt{ChiTietPhieuNhap}, \texttt{PhieuXuatKho}, \texttt{ChiTietPhieuXuat}.
    \item \textbf{Sơ đồ lớp kiểm kê, đơn hàng và thanh toán}: bao gồm các lớp liên quan đến kiểm kê, đơn đặt hàng, thanh toán và log thao tác: \texttt{PhieuKiemKe}, \texttt{ChiTietKiemKe}, \texttt{DonDatHang}, \texttt{ChiTietDonHang}, \texttt{ThanhToan}, \texttt{LichSuThaoTac}.
\end{itemize}

\section{Biểu đồ lớp trong ca sử dụng}

\subsection{UC05 -- Quản lý hàng hóa}

Ca sử dụng quản lý hàng hóa liên quan đến các lớp: \texttt{NguoiDung} (người thực hiện), \texttt{HangHoa} (đối tượng quản lý), \texttt{NhomHangHoa} (phân loại), \texttt{DonViTinh} (đơn vị đo) và \texttt{TonKho} (tồn kho tại các kho).

\textbf{Quan hệ giữa các lớp}:
\begin{itemize}[leftmargin=1.5cm]
    \item \texttt{NguoiDung} quản lý \texttt{HangHoa}: quan hệ 1..n.
    \item \texttt{HangHoa} thuộc \texttt{NhomHangHoa}: quan hệ n..1.
    \item \texttt{HangHoa} đo bằng \texttt{DonViTinh}: quan hệ n..1.
    \item \texttt{HangHoa} có \texttt{TonKho} tại nhiều \texttt{Kho}: quan hệ 1..n.
\end{itemize}

Khi thêm hàng hóa mới, hệ thống tự động tạo bản ghi \texttt{TonKho} bằng 0 tại tất cả các kho đang hoạt động. Khi cập nhật giá hoặc thông tin, chỉ bảng \texttt{HangHoa} thay đổi.

\subsection{UC09 -- Lập phiếu nhập kho}

Ca sử dụng lập phiếu nhập kho liên quan đến các lớp: \texttt{NguoiDung}, \texttt{PhieuNhapKho}, \texttt{ChiTietPhieuNhap}, \texttt{NhaCungCap}, \texttt{Kho}, \texttt{HangHoa} và \texttt{TonKho}.

\textbf{Quan hệ giữa các lớp}:
\begin{itemize}[leftmargin=1.5cm]
    \item \texttt{NguoiDung} lập \texttt{PhieuNhapKho}: quan hệ 1..n.
    \item \texttt{NhaCungCap} cung cấp cho \texttt{PhieuNhapKho}: quan hệ 1..n.
    \item \texttt{Kho} nhận \texttt{PhieuNhapKho}: quan hệ 1..n.
    \item \texttt{PhieuNhapKho} có nhiều \texttt{ChiTietPhieuNhap}: quan hệ 1..n.
    \item \texttt{ChiTietPhieuNhap} tham chiếu \texttt{HangHoa}: quan hệ n..1.
    \item Khi duyệt phiếu: \texttt{TonKho} được cập nhật (tăng số lượng).
\end{itemize}

Sau khi phiếu nhập được duyệt, lớp \texttt{TonKho} được cập nhật để phản ánh số lượng hàng mới nhập. Nếu bản ghi tồn kho chưa có, hệ thống tạo mới; nếu đã có, hệ thống cộng thêm số lượng.

\subsection{UC10 -- Lập phiếu xuất kho}

Ca sử dụng lập phiếu xuất kho liên quan đến các lớp: \texttt{NguoiDung}, \texttt{PhieuXuatKho}, \texttt{ChiTietPhieuXuat}, \texttt{KhachHang}, \texttt{Kho}, \texttt{HangHoa} và \texttt{TonKho}.

\textbf{Quan hệ giữa các lớp}:
\begin{itemize}[leftmargin=1.5cm]
    \item \texttt{NguoiDung} lập \texttt{PhieuXuatKho}: quan hệ 1..n.
    \item \texttt{KhachHang} nhận \texttt{PhieuXuatKho}: quan hệ 1..n.
    \item \texttt{Kho} xuất \texttt{PhieuXuatKho}: quan hệ 1..n.
    \item \texttt{PhieuXuatKho} có nhiều \texttt{ChiTietPhieuXuat}: quan hệ 1..n.
    \item \texttt{ChiTietPhieuXuat} tham chiếu \texttt{HangHoa}: quan hệ n..1.
    \item Khi duyệt phiếu: \texttt{TonKho} được cập nhật (giảm số lượng).
\end{itemize}

Hệ thống kiểm tra \texttt{TonKho} trước khi cho phép xuất: nếu số lượng tồn tại kho không đủ, hệ thống từ chối thêm dòng hàng đó. Sau khi duyệt, tồn kho được giảm tương ứng.

\section{Kết luận chương}

Chương 4 đã trình bày sơ đồ lớp tổng thể với 18 lớp chính của hệ thống, mô tả chi tiết thuộc tính và quan hệ giữa các lớp. Đồng thời phân tích sâu ba biểu đồ lớp trong ca sử dụng quan trọng: UC05 (quản lý hàng hóa), UC09 (lập phiếu nhập kho) và UC10 (lập phiếu xuất kho). Các lớp và quan hệ được xác định là cơ sở để thiết kế tương tác (Sequence Diagram) trong Chương 5 và thiết kế chi tiết trong Chương 6, 7.
