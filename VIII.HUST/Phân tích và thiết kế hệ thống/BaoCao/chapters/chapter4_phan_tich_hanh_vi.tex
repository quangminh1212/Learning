\chapter{PHÂN TÍCH CẤU TRÚC}

\section{Giới thiệu giai đoạn}

\textbf{Phân tích cấu trúc} là giai đoạn thứ hai trong ba giai đoạn phân tích hướng đối tượng. Giai đoạn này trả lời: hệ thống được cấu thành từ những \textbf{lớp đối tượng} nào? Mỗi lớp có thuộc tính gì? Quan hệ giữa các lớp ra sao?

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Biểu đồ chính}: Class Diagram (biểu đồ lớp).
    \item \textbf{Trước đó}: Ch.3 Phân tích chức năng -- Use Case (đã biết actor và mục tiêu).
    \item \textbf{Sau đó}: Ch.5 Phân tích hành vi -- Sequence (các đối tượng tương tác thế nào theo thời gian).
\end{itemize}

Từ đặc tả use case, các danh từ nghiệp vụ quan trọng được ứng viên thành lớp (ví dụ: hàng hóa, phiếu nhập, tồn kho, người dùng\ldots). Các lớp liên quan chặt được nhóm thành sơ đồ con để dễ đọc.

\section{Biểu đồ lớp trong ca sử dụng}

Phân tích cấu trúc đi sâu vào một số use case quan trọng: chỉ ra \textbf{các lớp tham gia} và \textbf{quan hệ} cần có để hoàn thành mục tiêu use case.

\subsection{UC03 -- Thêm tài khoản}

\textbf{Mục tiêu use case}: tạo bản ghi người dùng mới.

\textbf{Lớp tham gia}: \texttt{NguoiDung} (đối tượng được tạo/cập nhật), \texttt{LichSuThaoTac} (ghi nhận thao tác thêm -- tùy chọn thiết kế).

\textbf{Quan hệ}: Một \texttt{NguoiDung} có thể sinh ra nhiều bản ghi lịch sử thao tác (1..n).

\input{diagrams/tex/6.2_class_uc03}

\subsection{UC06 -- Thêm hàng hóa}

\textbf{Mục tiêu use case}: thêm mặt hàng và khởi tạo tồn.

\textbf{Lớp tham gia}:
\begin{itemize}[leftmargin=1.5cm]
    \item \texttt{NguoiDung}: người thực hiện.
    \item \texttt{HangHoa}: đối tượng được tạo.
    \item \texttt{NhomHangHoa}, \texttt{DonViTinh}: tham chiếu bắt buộc khi tạo hàng.
    \item \texttt{TonKho}, \texttt{Kho}: khởi tạo tồn = 0 tại từng kho.
\end{itemize}

\textbf{Quan hệ chính}: HangHoa thuộc 1 NhomHangHoa; đo bằng 1 DonViTinh; có nhiều TonKho (theo từng Kho).

\input{diagrams/tex/4.4_class_uc05}

\subsection{UC09 -- Lập phiếu nhập kho}

\textbf{Mục tiêu use case}: tạo phiếu nhập, chi tiết dòng hàng; khi duyệt cập nhật tồn.

\textbf{Lớp tham gia}: \texttt{NguoiDung}, \texttt{PhieuNhapKho}, \texttt{ChiTietPhieuNhap}, \texttt{NhaCungCap}, \texttt{Kho}, \texttt{HangHoa}, \texttt{TonKho}.

\textbf{Quan hệ chính}:
\begin{itemize}[leftmargin=1.5cm]
    \item Một phiếu nhập có nhiều chi tiết (1..n).
    \item Mỗi chi tiết tham chiếu một HangHoa.
    \item Phiếu gắn một NhaCungCap, một Kho, một NguoiDung (người lập).
    \item Khi duyệt: TonKho tại (Kho, HangHoa) được tăng.
\end{itemize}

\subsection{UC10 -- Lập phiếu xuất kho}

\textbf{Mục tiêu use case}: tạo phiếu xuất; kiểm tra tồn khi thêm dòng; duyệt thì giảm tồn.

\textbf{Lớp tham gia}: \texttt{NguoiDung}, \texttt{PhieuXuatKho}, \texttt{ChiTietPhieuXuat}, \texttt{KhachHang}, \texttt{Kho}, \texttt{HangHoa}, \texttt{TonKho}.

\textbf{Quan hệ}: tương tự phiếu nhập, thay NhaCungCap bằng KhachHang; TonKho dùng để kiểm tra và giảm số lượng.
