\chapter{PHÂN TÍCH HỆ THỐNG}

\section{Mô tả nghiệp vụ}

Hệ thống quản lý kho hàng đảm nhận việc kiểm soát toàn bộ vòng đời của hàng hóa từ khi nhập vào kho cho đến khi xuất ra khỏi kho. Các nghiệp vụ chính bao gồm:

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Nhập kho}: Tiếp nhận hàng hóa từ nhà cung cấp, kiểm tra chất lượng, số lượng và ghi nhận vào hệ thống.
    \item \textbf{Xuất kho}: Lập phiếu xuất kho cho khách hàng hoặc sản xuất, cập nhật số lượng tồn kho.
    \item \textbf{Kiểm kê}: Định kỳ kiểm tra thực tế hàng tồn kho so với sổ sách, điều chỉnh chênh lệch nếu có.
    \item \textbf{Quản lý hàng hóa}: Cập nhật thông tin hàng hóa, nhóm hàng, đơn vị tính, giá nhập/xuất.
    \item \textbf{Báo cáo}: Tổng hợp số liệu nhập xuất tồn, hàng tồn lâu ngày, hàng sắp hết.
\end{itemize}

\section{Phân tích nghiệp vụ hiện trạng (As-Is)}

Hiện tại, nhiều doanh nghiệp vẫn quản lý kho bằng các phương pháp thủ công. Quy trình điển hình như sau:

\begin{enumerate}[leftmargin=1.5cm]
    \item Nhân viên kho nhận hàng, đối chiếu với chứng từ giấy từ nhà cung cấp.
    \item Số liệu được ghi vào sổ nhập/xuất hoặc nhập tay vào file Excel.
    \item Quản lý kho tổng hợp số liệu cuối ngày/tuần/tháng để tính tồn kho.
    \item Báo cáo được lập thủ công, dễ sai sót và không cập nhật theo thời gian thực.
\end{enumerate}

Nhược điểm của hiện trạng:
\begin{itemize}[leftmargin=1.5cm]
    \item Dữ liệu phân tán, khó tổng hợp và đối chiếu.
    \item Thiếu cơ chế kiểm soát số lượng tồn kho theo thời gian thực.
    \item Khó truy vết lịch sử nhập xuất và người thực hiện thao tác.
    \item Báo cáo chậm, không hỗ trợ ra quyết định kịp thời.
\end{itemize}

\section{Các yêu cầu chức năng}

Dựa trên phân tích nghiệp vụ, hệ thống cần đáp ứng các yêu cầu chức năng sau:

\begin{enumerate}[leftmargin=1.5cm]
    \item Quản lý danh mục hàng hóa, nhóm hàng, đơn vị tính, nhà cung cấp, khách hàng.
    \item Quản lý phiếu nhập kho và chi tiết nhập kho.
    \item Quản lý phiếu xuất kho và chi tiết xuất kho.
    \item Quản lý điều chuyển kho nội bộ.
    \item Quản lý tồn kho theo từng kho, từng sản phẩm.
    \item Kiểm kê định kỳ và điều chỉnh tồn kho.
    \item Cảnh báo tồn kho thấp, tồn kho cao, hàng hết hạn.
    \item Quản lý người dùng và phân quyền.
    \item Báo cáo nhập xuất tồn, báo cáo hàng hóa, báo cáo tài chính tồn kho.
\end{enumerate}

\section{Các yêu cầu phi chức năng}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Hiệu năng}: Hệ thống phản hồi các thao tác tra cứu và cập nhật trong vòng vài giây.
    \item \textbf{Bảo mật}: Phân quyền rõ ràng, mã hóa mật khẩu, ghi log thao tác.
    \item \textbf{Khả dụng}: Hệ thống hoạt động ổn định, sao lưu dữ liệu định kỳ.
    \item \textbf{Khả mở rộng}: Dễ dàng bổ sung thêm kho, chi nhánh, loại hàng hóa.
    \item \textbf{Giao diện}: Thân thiện, dễ sử dụng, hỗ trợ tiếng Việt.
\end{itemize}

\section{Danh sách use case}

Các use case chính của hệ thống được nhóm theo module:

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Module Quản lý hàng hóa}: Thêm hàng hóa, cập nhật hàng hóa, xóa hàng hóa, tìm kiếm hàng hóa.
    \item \textbf{Module Quản lý nhập kho}: Lập phiếu nhập kho, duyệt phiếu nhập, nhập hàng vào kho, in phiếu nhập.
    \item \textbf{Module Quản lý xuất kho}: Lập phiếu xuất kho, duyệt phiếu xuất, xuất hàng khỏi kho, in phiếu xuất.
    \item \textbf{Module Quản lý tồn kho}: Xem tồn kho, kiểm kê, điều chỉnh tồn kho, cảnh báo tồn kho.
    \item \textbf{Module Quản lý người dùng}: Đăng nhập, đăng xuất, phân quyền, quản lý tài khoản.
    \item \textbf{Module Báo cáo}: Báo cáo nhập xuất tồn, báo cáo tồn kho, báo cáo hàng hóa.
\end{itemize}

\section{Sơ đồ use case phân rã}

Các use case phân rã sẽ được trình bày chi tiết trong phần thiết kế. Các nhóm use case chính bao gồm:

\begin{itemize}[leftmargin=1.5cm]
    \item Phân rã use case Quản lý nhập kho.
    \item Phân rã use case Quản lý xuất kho.
    \item Phân rã use case Quản lý tồn kho.
    \item Phân rã use case Quản lý hàng hóa.
\end{itemize}

\section{Đặc tả use case}

\subsection{Đặc tả use case ``Đăng nhập''}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Quản trị viên, Quản lý kho, Nhân viên kho.
    \item \textbf{Mục đích}: Xác thực người dùng trước khi truy cập hệ thống.
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item Người dùng truy cập giao diện đăng nhập.
        \item Người dùng nhập tên đăng nhập và mật khẩu.
        \item Hệ thống kiểm tra thông tin.
        \item Hệ thống chuyển người dùng đến trang chủ phù hợp với quyền hạn.
    \end{enumerate}
    \item \textbf{Luồng thay thế}: Nếu thông tin sai, hệ thống thông báo lỗi và yêu cầu nhập lại.
\end{itemize}

\subsection{Đặc tả use case ``Nhập kho''}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho, Quản lý kho.
    \item \textbf{Mục đích}: Ghi nhận hàng hóa nhập vào kho.
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item Nhân viên kho chọn chức năng lập phiếu nhập kho.
        \item Hệ thống hiển thị form nhập liệu.
        \item Nhân viên kho chọn nhà cung cấp, kho nhập, ngày nhập.
        \item Nhân viên kho thêm các dòng chi tiết hàng hóa nhập.
        \item Hệ thống kiểm tra số lượng tồn kho cho phép và tính hợp lệ.
        \item Nhân viên kho gửi phiếu nhập để quản lý kho duyệt.
        \item Sau khi duyệt, hệ thống cập nhật tồn kho.
    \end{enumerate}
\end{itemize}

\subsection{Đặc tả use case ``Xuất kho''}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho, Quản lý kho.
    \item \textbf{Mục đích}: Ghi nhận hàng hóa xuất ra khỏi kho.
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item Nhân viên kho chọn chức năng lập phiếu xuất kho.
        \item Hệ thống hiển thị form nhập liệu.
        \item Nhân viên kho chọn khách hàng, kho xuất, ngày xuất.
        \item Nhân viên kho thêm các dòng chi tiết hàng hóa xuất.
        \item Hệ thống kiểm tra số lượng tồn kho đủ để xuất.
        \item Nhân viên kho gửi phiếu xuất để quản lý kho duyệt.
        \item Sau khi duyệt, hệ thống giảm tồn kho tương ứng.
    \end{enumerate}
\end{itemize}

\subsection{Đặc tả use case ``Kiểm kê tồn kho''}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho, Quản lý kho.
    \item \textbf{Mục đích}: Đối chiếu tồn kho thực tế với sổ sách và điều chỉnh chênh lệch.
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item Nhân viên kho chọn kho và danh sách hàng cần kiểm kê.
        \item Nhân viên kho nhập số lượng thực tế cho từng hàng hóa.
        \item Hệ thống tính chênh lệch so với sổ sách.
        \item Quản lý kho duyệt phiếu kiểm kê.
        \item Hệ thống cập nhật tồn kho theo số lượng thực tế.
    \end{enumerate}
\end{itemize}
