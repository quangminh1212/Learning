\chapter{MÔ TẢ NGHIỆP VỤ}

\section{Quy trình phân tích -- thiết kế (hướng đối tượng)}

Tên các chương phân tích theo \textbf{nội dung phân tích}, còn biểu đồ chỉ là \textbf{công cụ}:

\begin{enumerate}[leftmargin=1.5cm]
    \item \textbf{Phân tích chức năng} (Ch.3) -- dùng biểu đồ Use Case + đặc tả.
    \item \textbf{Phân tích cấu trúc} (Ch.4) -- dùng biểu đồ Class.
    \item \textbf{Phân tích hành vi} (Ch.5) -- dùng biểu đồ Sequence.
\end{enumerate}

Chương này: mô tả \textbf{nghiệp vụ} + biểu đồ hoạt động. Sau ba giai đoạn phân tích: thiết kế (Ch.6--7).

\textit{Lưu ý}:
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Nghiệp vụ} = công việc nghiệp vụ của doanh nghiệp (nhập kho, xuất kho, kiểm kê\ldots).
    \item \textbf{Không} coi \textbf{đăng nhập / đăng xuất} là nghiệp vụ -- đó là \textbf{chức năng hệ thống} (hỗ trợ an toàn truy cập), được mô hình hóa bằng use case ở Chương 3 (UC01, UC02).
    \item Nghiệp vụ ``Quản lý \ldots'' là gói chung; use case = một mục tiêu cụ thể của actor.
\end{itemize}

\section{Nghiệp vụ 1: Quản lý tài khoản}

\textbf{Mô tả}: Quản trị viên duy trì danh sách tài khoản trên hệ thống, gán vai trò (quản trị viên, quản lý kho, nhân viên kho) để phân quyền. Chỉ người được cấp quyền tương ứng mới thực hiện được các thao tác nhập/xuất, duyệt phiếu, quản lý danh mục.

\textbf{Người thực hiện}: Quản trị viên.

\textbf{Quy trình} (thao tác tiêu biểu -- thêm tài khoản):
\begin{enumerate}[leftmargin=1.5cm]
    \item Quản trị viên đăng nhập và chọn chức năng quản lý tài khoản.
    \item Chọn ``Thêm tài khoản'', nhập họ tên, tên đăng nhập, mật khẩu, email (nếu có), vai trò, trạng thái.
    \item Hệ thống kiểm tra tên đăng nhập chưa tồn tại.
    \item Nếu hợp lệ: mã hóa mật khẩu, lưu vào cơ sở dữ liệu, hiển thị thông báo thành công.
    \item Nếu trùng tên đăng nhập: báo lỗi, yêu cầu đổi tên.
\end{enumerate}

Ngoài ra, quản trị viên có thể \textbf{sửa} thông tin/vai trò, \textbf{khóa/mở} tài khoản, \textbf{tìm kiếm} danh sách -- luồng tương tự (chọn bản ghi $\rightarrow$ thực hiện $\rightarrow$ lưu).

\textbf{Dữ liệu liên quan}: \texttt{NguoiDung} (mã, tên đăng nhập, mật khẩu, họ tên, vai trò, trạng thái).

\textbf{Use case tiêu biểu}: UC03 Thêm tài khoản.

\section{Nghiệp vụ 2: Quản lý nhà cung cấp}

\textbf{Mô tả}: Lưu trữ và duy trì thông tin các nhà cung cấp hàng hóa (tên, địa chỉ, điện thoại, email, người liên hệ). Thông tin NCC được dùng khi lập phiếu nhập kho, giúp truy vết hàng đến từ ai.

\textbf{Người thực hiện}: Nhân viên kho, Quản lý kho.

\textbf{Quy trình} (thao tác tiêu biểu -- thêm nhà cung cấp):
\begin{enumerate}[leftmargin=1.5cm]
    \item Người dùng chọn chức năng quản lý nhà cung cấp, xem danh sách hiện có.
    \item Chọn ``Thêm mới'', nhập mã NCC, tên, địa chỉ, số điện thoại, email, người liên hệ, mô tả (nếu có).
    \item Hệ thống kiểm tra mã NCC không trùng.
    \item Lưu vào cơ sở dữ liệu; có thể sửa hoặc đánh dấu ngừng hoạt động nếu NCC không còn giao dịch (không xóa cứng khi đã có phiếu nhập liên quan).
\end{enumerate}

\textbf{Dữ liệu liên quan}: \texttt{NhaCungCap} (mã, tên, địa chỉ, SĐT, email, người liên hệ, mô tả).

\textbf{Use case tiêu biểu}: UC04 Thêm nhà cung cấp.

\section{Nghiệp vụ 3: Quản lý khách hàng}

\textbf{Mô tả}: Tương tự nhà cung cấp, nghiệp vụ này quản lý danh mục khách hàng nhận hàng khi xuất kho (tên, địa chỉ, liên hệ, mã số thuế nếu là doanh nghiệp). Phục vụ lập phiếu xuất và theo dõi lịch sử giao dịch theo khách.

\textbf{Người thực hiện}: Nhân viên kho, Quản lý kho.

\textbf{Quy trình} (thao tác tiêu biểu -- thêm khách hàng):
\begin{enumerate}[leftmargin=1.5cm]
    \item Người dùng chọn quản lý khách hàng, xem danh sách.
    \item Thêm mới: nhập mã KH, tên, địa chỉ, SĐT, email, mã số thuế, người liên hệ.
    \item Hệ thống kiểm tra mã không trùng và lưu.
    \item Có thể sửa thông tin; nếu đã có phiếu xuất liên quan thì không xóa cứng, chỉ ngừng sử dụng.
\end{enumerate}

\textbf{Dữ liệu liên quan}: \texttt{KhachHang} (mã, tên, địa chỉ, SĐT, email, MST, người liên hệ).

\textbf{Use case tiêu biểu}: UC05 Thêm khách hàng.

\section{Nghiệp vụ 4: Quản lý hàng hóa}

\textbf{Mô tả}: Duy trì danh mục hàng hóa trong hệ thống: mã, tên, nhóm hàng, đơn vị tính, giá nhập, giá xuất, mô tả, trạng thái. Hàng hóa là trung tâm của hệ thống -- mọi phiếu nhập/xuất và tồn kho đều tham chiếu đến mã hàng. Khi thêm hàng mới, hệ thống tự khởi tạo số lượng tồn bằng 0 tại các kho đang hoạt động.

\textbf{Người thực hiện}: Nhân viên kho, Quản lý kho.

\textbf{Quy trình thêm hàng hóa}:
\begin{enumerate}[leftmargin=1.5cm]
    \item Chọn quản lý hàng hóa, chọn ``Thêm mới''.
    \item Nhập mã hàng, tên; chọn nhóm hàng và đơn vị tính; nhập giá nhập, giá xuất, mô tả.
    \item Hệ thống kiểm tra mã không trùng, nhóm và ĐVT hợp lệ.
    \item Lưu bản ghi \texttt{HangHoa}; tạo các bản ghi \texttt{TonKho} với số lượng = 0 cho từng kho.
\end{enumerate}

\textbf{Quy trình tìm kiếm hàng hóa}:
\begin{enumerate}[leftmargin=1.5cm]
    \item Người dùng nhập từ khóa (mã hoặc tên) và/hoặc chọn bộ lọc (nhóm hàng, kho, trạng thái).
    \item Hệ thống truy vấn, trả về danh sách hàng kèm số lượng tồn tại các kho.
    \item Người dùng có thể xem chi tiết, sắp xếp hoặc xuất danh sách.
\end{enumerate}

Có thể sửa thông tin/giá hoặc đánh dấu ngừng sử dụng hàng không còn kinh doanh (không xóa nếu đã phát sinh phiếu).

\textbf{Dữ liệu liên quan}: \texttt{HangHoa}, \texttt{NhomHangHoa}, \texttt{DonViTinh}, \texttt{TonKho}, \texttt{Kho}.

\textbf{Use case tiêu biểu}: UC06 Thêm hàng hóa; UC07 Tìm kiếm hàng hóa.

\section{Nghiệp vụ 5: Quản lý kho}

\textbf{Mô tả}: Doanh nghiệp có thể có nhiều kho (địa điểm khác nhau). Nghiệp vụ này duy trì thông tin từng kho: mã, tên, địa chỉ, người phụ trách, diện tích, trạng thái hoạt động. Tồn kho được theo dõi theo từng cặp (kho, hàng hóa).

\textbf{Người thực hiện}: Quản lý kho, Quản trị viên.

\textbf{Quy trình} (thao tác tiêu biểu -- thêm kho):
\begin{enumerate}[leftmargin=1.5cm]
    \item Người có quyền chọn quản lý kho, chọn ``Thêm mới''.
    \item Nhập mã kho, tên kho, địa chỉ, người phụ trách, diện tích, mô tả.
    \item Hệ thống kiểm tra mã kho không trùng và lưu.
    \item Có thể sửa thông tin hoặc đánh dấu kho ngừng hoạt động.
    \item Tra cứu kho: xem thông tin kho và danh sách hàng đang tồn trong kho đó.
\end{enumerate}

\textbf{Dữ liệu liên quan}: \texttt{Kho}, \texttt{TonKho}, \texttt{HangHoa}.

\textbf{Use case tiêu biểu}: UC08 Thêm kho.

\section{Nghiệp vụ 6: Nhập kho}

\textbf{Mô tả}: Khi hàng từ nhà cung cấp về kho, nhân viên lập \textbf{phiếu nhập kho} ghi nhận NCC, kho nhận, ngày nhập và từng dòng hàng (mã hàng, số lượng, đơn giá). Phiếu được gửi \textbf{duyệt}; chỉ khi quản lý kho duyệt, hệ thống mới \textbf{tăng tồn kho} và khóa phiếu. Nếu từ chối, tồn không đổi.

\textbf{Người thực hiện}: Nhân viên kho (lập phiếu); Quản lý kho (duyệt phiếu).

\textbf{Quy trình}:
\begin{enumerate}[leftmargin=1.5cm]
    \item Nhân viên kho chọn lập phiếu nhập: chọn nhà cung cấp, kho nhập, ngày nhập, ghi chú.
    \item Thêm từng dòng chi tiết: chọn hàng hóa, nhập số lượng, đơn giá; hệ thống tính thành tiền và tổng tiền phiếu.
    \item Kiểm tra dữ liệu hợp lệ (đã chọn NCC, kho, có ít nhất một dòng hàng).
    \item Gửi duyệt -- trạng thái phiếu: ``Chờ duyệt''.
    \item Quản lý kho xem phiếu, kiểm tra và \textbf{duyệt} hoặc \textbf{từ chối} (kèm lý do).
    \item Nếu duyệt: trạng thái ``Đã duyệt''; với mỗi dòng hàng, \texttt{TonKho} tại kho nhập được \textbf{cộng} số lượng; ghi log thao tác.
    \item Nếu từ chối: trạng thái ``Từ chối''; tồn kho không thay đổi.
\end{enumerate}

\textbf{Dữ liệu liên quan}: \texttt{PhieuNhapKho}, \texttt{ChiTietPhieuNhap}, \texttt{TonKho}, \texttt{NhaCungCap}, \texttt{Kho}, \texttt{HangHoa}, \texttt{LichSuThaoTac}.

\textbf{Use case tiêu biểu}: UC09 Lập phiếu nhập kho (bao gồm luồng lập và duyệt với hai actor).

\section{Nghiệp vụ 7: Xuất kho}

\textbf{Mô tả}: Khi xuất hàng cho khách hàng hoặc sử dụng nội bộ, nhân viên lập \textbf{phiếu xuất kho}. Hệ thống \textbf{kiểm tra tồn đủ} trước khi cho phép thêm dòng hàng. Sau khi quản lý duyệt, tồn kho được \textbf{giảm} tương ứng.

\textbf{Người thực hiện}: Nhân viên kho (lập phiếu); Quản lý kho (duyệt phiếu).

\textbf{Quy trình}:
\begin{enumerate}[leftmargin=1.5cm]
    \item Nhân viên kho chọn lập phiếu xuất: chọn khách hàng, kho xuất, ngày xuất, lý do xuất.
    \item Thêm dòng hàng: chọn hàng, nhập số lượng. Hệ thống kiểm tra \texttt{TonKho} tại kho xuất.
    \item Nếu tồn \textbf{không đủ}: thông báo, không cho thêm dòng đó.
    \item Nếu đủ: hiển thị đơn giá xuất, tính thành tiền và tổng tiền phiếu.
    \item Gửi duyệt -- trạng thái ``Chờ duyệt''.
    \item Quản lý kho duyệt: trạng thái ``Đã duyệt''; \textbf{giảm} tồn kho từng dòng; ghi log.
    \item Nếu từ chối: tồn không đổi, ghi lý do từ chối.
\end{enumerate}

\textbf{Dữ liệu liên quan}: \texttt{PhieuXuatKho}, \texttt{ChiTietPhieuXuat}, \texttt{TonKho}, \texttt{KhachHang}, \texttt{Kho}, \texttt{HangHoa}, \texttt{LichSuThaoTac}.

\textbf{Use case tiêu biểu}: UC10 Lập phiếu xuất kho (lập + duyệt).

\section{Nghiệp vụ 8: Kiểm kê tồn kho}

\textbf{Mô tả}: Định kỳ (tháng/quý), đối chiếu số lượng hàng \textbf{thực tế} trong kho với số lượng trên hệ thống (sổ sách). Nếu có chênh lệch, quản lý duyệt điều chỉnh để \texttt{TonKho} khớp thực tế.

\textbf{Người thực hiện}: Nhân viên kho (lập phiếu, nhập số thực tế); Quản lý kho (duyệt điều chỉnh).

\textbf{Quy trình}:
\begin{enumerate}[leftmargin=1.5cm]
    \item Nhân viên tạo phiếu kiểm kê: chọn kho, ngày kiểm.
    \item Hệ thống hiển thị danh sách hàng và \textbf{số lượng theo sổ} (từ \texttt{TonKho}).
    \item Nhân viên kiểm đếm thực tế, nhập \textbf{số lượng thực tế} từng mặt hàng.
    \item Hệ thống tính \textbf{chênh lệch} = thực tế $-$ sổ sách.
    \item Gửi duyệt. Quản lý xem chênh lệch, \textbf{duyệt điều chỉnh} hoặc yêu cầu kiểm lại.
    \item Nếu duyệt: cập nhật \texttt{TonKho} = số thực tế; ghi log; trạng thái phiếu ``Đã duyệt''.
\end{enumerate}

\textbf{Dữ liệu liên quan}: \texttt{PhieuKiemKe}, \texttt{ChiTietKiemKe}, \texttt{TonKho}, \texttt{Kho}, \texttt{HangHoa}, \texttt{LichSuThaoTac}.

\textbf{Use case tiêu biểu}: UC11 Lập phiếu kiểm kê.

\section{Nghiệp vụ 9: Báo cáo nhập -- xuất -- tồn}

\textbf{Mô tả}: Quản lý kho / quản trị viên xem \textbf{báo cáo tồn kho} và tổng hợp nhập--xuất--tồn theo khoảng thời gian, theo kho hoặc theo hàng, phục vụ theo dõi vận hành (không bao gồm module tài chính/thanh toán phức tạp).

\textbf{Người thực hiện}: Quản lý kho, Quản trị viên.

\textbf{Quy trình}:
\begin{enumerate}[leftmargin=1.5cm]
    \item Chọn loại xem: tồn hiện tại, hoặc nhập--xuất--tồn theo kỳ.
    \item Nhập điều kiện: khoảng ngày, kho (tùy chọn), nhóm hàng / mã hàng (tùy chọn).
    \item Hệ thống tổng hợp từ \texttt{PhieuNhapKho}, \texttt{PhieuXuatKho}, \texttt{TonKho}, \texttt{HangHoa}.
    \item Hiển thị bảng kết quả; có thể xuất Excel để lưu trữ.
\end{enumerate}

\textbf{Dữ liệu liên quan}: \texttt{PhieuNhapKho}, \texttt{ChiTietPhieuNhap}, \texttt{PhieuXuatKho}, \texttt{ChiTietPhieuXuat}, \texttt{TonKho}, \texttt{HangHoa}, \texttt{Kho}.

\textbf{Use case tiêu biểu}: UC12 Xem báo cáo nhập xuất tồn.

\section{Biểu đồ hoạt động (theo nghiệp vụ chính)}

Biểu đồ hoạt động minh họa luồng thực hiện các nghiệp vụ nhập, xuất, kiểm kê (trước khi vào 3 giai đoạn phân tích UML).

\subsection{Nghiệp vụ nhập kho}
\input{diagrams/tex/2.1_activity_nhapkho}

Giải thích sơ đồ: Biểu đồ hoạt động mô tả luồng \textbf{nhập kho} từ lúc bắt đầu đến kết thúc. Các hình chữ nhật là bước xử lý (lập phiếu, nhập NCC/kho, thêm chi tiết hàng). Hình thoi là điểm quyết định: (1) dữ liệu phiếu có hợp lệ không -- nếu không thì báo lỗi và quay lại nhập; (2) quản lý có duyệt không -- nếu duyệt thì cập nhật tồn kho và in/lưu phiếu, nếu từ chối thì thông báo và có thể lập lại. Mũi tên thể hiện thứ tự thực hiện; nhánh ``Không'' quay vòng để sửa sai trước khi đi tiếp.

\subsection{Nghiệp vụ xuất kho}
\input{diagrams/tex/2.2_activity_xuatkho}

Giải thích sơ đồ: Tương tự nhập kho nhưng gắn nghiệp vụ \textbf{xuất hàng}. Luồng đi qua lập phiếu, chọn khách/kho, thêm dòng hàng (kèm kiểm tra tồn đủ), gửi duyệt, quản lý duyệt/từ chối. Khác biệt cốt lõi so với nhập: tại bước thêm hàng, hệ thống phải \textbf{chặn xuất vượt tồn}; khi duyệt thành công thì \textbf{giảm tồn} thay vì tăng. Các quyết định ``hợp lệ?'' / ``duyệt?'' tạo nhánh xử lý lỗi và từ chối rõ ràng.

Giải thích sơ đồ: Biểu đồ thể hiện quy trình \textbf{kiểm kê định kỳ}: tạo phiếu theo kho $\rightarrow$ hệ thống nạp số lượng theo sổ $\rightarrow$ nhân viên nhập số thực tế $\rightarrow$ tính chênh lệch $\rightarrow$ gửi duyệt $\rightarrow$ quản lý duyệt điều chỉnh thì cập nhật tồn bằng số thực tế. Sơ đồ nhấn mạnh rằng tồn kho trên hệ thống chỉ đổi \textbf{sau khi duyệt}, không đổi ngay khi nhập số liệu kiểm đếm.
