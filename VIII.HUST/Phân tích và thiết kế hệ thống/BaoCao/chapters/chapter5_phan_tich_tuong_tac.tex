\chapter{SƠ ĐỒ TRÌNH TỰ (SEQUENCE DIAGRAM)}

Sau activity, \textbf{Sequence diagram} mô tả tương tác giữa các đối tượng (actor, giao diện, controller, DAO, CSDL) theo thời gian để hoàn thành use case.

\section{UC15 -- Thêm hàng hóa}

\textbf{Mục tiêu}: Thêm một mặt hàng mới.

\textbf{Đối tượng}: NV kho, Giao diện, HangHoaController, HangHoaDAO, TonKhoDAO, CSDL.

\textbf{Luồng}:
\begin{enumerate}[leftmargin=1.5cm]
    \item NV gửi ``Thêm hàng hóa'' qua giao diện.
    \item Giao diện $\rightarrow$ HangHoaController.
    \item Controller $\rightarrow$ HangHoaDAO kiểm tra mã trùng.
    \item Không trùng: DAO lưu HangHoa.
    \item Controller $\rightarrow$ TonKhoDAO khởi tạo tồn = 0 các kho.
    \item Trả thông báo thành công ra giao diện.
\end{enumerate}

\section{UC22 -- Lập phiếu nhập kho}

\textbf{Mục tiêu}: Lập phiếu + gửi duyệt (tăng tồn khi UC23).

\textbf{Đối tượng}: NV, (QL ở bước duyệt), Giao diện, PhieuNhapController, PhieuNhapDAO, ChiTietPhieuNhapDAO, TonKhoDAO, LichSuDAO, CSDL.

\textbf{Luồng}:
\begin{enumerate}[leftmargin=1.5cm]
    \item NV tạo phiếu, thêm chi tiết dòng hàng.
    \item Gửi duyệt (trạng thái Chờ duyệt).
    \item QL duyệt $\rightarrow$ cập nhật trạng thái, tăng tồn, ghi log (UC23 trên cùng sequence tổng quát).
\end{enumerate}
