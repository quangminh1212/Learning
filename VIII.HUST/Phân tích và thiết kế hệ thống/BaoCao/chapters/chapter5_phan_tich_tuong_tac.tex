\chapter{PHÂN TÍCH HÀNH VI}

\section{Phân tích hành vi bằng biểu đồ trình tự}

Sau phân tích chức năng (use case) và cấu trúc (lớp), chương này dùng \textbf{biểu đồ trình tự (Sequence)} để mô tả các đối tượng tương tác theo thời gian khi thực hiện use case. Mỗi sơ đồ gắn một use case đã đặc tả: actor, giao diện, phần xử lý và dữ liệu.

\subsection{UC01 -- Đăng nhập}

\textbf{Bối cảnh}: Actor muốn vào hệ thống. Đây là hành vi xác thực -- điều kiện để thực hiện các use case nghiệp vụ.

\textbf{Đối tượng tham gia}:
\begin{itemize}[leftmargin=1.5cm]
    \item Người dùng (actor);
    \item Giao diện đăng nhập (boundary);
    \item Thành phần xử lý xác thực (control/hệ thống);
    \item Cơ sở dữ liệu / lớp \texttt{NguoiDung} (entity).
\end{itemize}

\textbf{Kịch bản chính}: Mở form $\rightarrow$ nhập tên/mật khẩu $\rightarrow$ hệ thống kiểm tra CSDL $\rightarrow$ tạo phiên hoặc báo lỗi $\rightarrow$ chuyển trang chủ nếu thành công.

\input{diagrams/tex/5.7_sequence_uc01}

Giải thích sơ đồ: Trục đứng là các đối tượng (Người dùng, Giao diện đăng nhập, Hệ thống, CSDL); trục thời gian đi từ trên xuống. Mũi tên ngang là thông điệp: (1)--(2) actor thao tác form; (3) giao diện gửi thông tin đăng nhập cho hệ thống; (4) hệ thống hỏi CSDL kiểm tra tài khoản; (5)--(6) trả kết quả và hiển thị trang chủ hoặc lỗi. Sơ đồ chứng minh hành vi UC01 là chuỗi tương tác \textit{boundary $\rightarrow$ control $\rightarrow$ entity}, không chỉ là ``màn hình login''.

\subsection{UC13 -- Thêm hàng hóa}

\textbf{Bối cảnh}: NV/QL thêm mặt hàng mới; hệ thống phải vừa lưu danh mục vừa khởi tạo tồn.

\textbf{Đối tượng tham gia}: Nhân viên kho, giao diện hàng hóa, xử lý nghiệp vụ hàng hóa, CSDL (\texttt{HangHoa}, \texttt{TonKho}).

\textbf{Kịch bản chính}:
\begin{enumerate}[leftmargin=1.5cm]
    \item Actor chọn thêm hàng, nhập thông tin trên form.
    \item Giao diện gửi dữ liệu cho hệ thống.
    \item Hệ thống kiểm tra mã/trùng/lệ thuộc nhóm--ĐVT.
    \item Lưu \texttt{HangHoa}; tạo \texttt{TonKho} = 0 các kho.
    \item Trả thông báo và làm mới danh sách.
\end{enumerate}

\input{diagrams/tex/5.1_sequence_uc05}

Giải thích sơ đồ: Sequence thể hiện thứ tự: NV chọn quản lý hàng hóa $\rightarrow$ nhập thông tin $\rightarrow$ giao diện gửi hệ thống $\rightarrow$ kiểm tra dữ liệu $\rightarrow$ lưu \texttt{HangHoa} xuống CSDL $\rightarrow$ cập nhật/tạo tồn ban đầu $\rightarrow$ trả thông báo thành công $\rightarrow$ hiển thị lại danh sách. Các lời gọi lồng nhau (hệ thống $\rightarrow$ CSDL) cho thấy bước lưu và khởi tạo tồn là hai thao tác dữ liệu tuần tự sau khi validate.

\subsection{UC16 -- Tìm kiếm hàng hóa}

\textbf{Bối cảnh}: Tra cứu không làm thay đổi dữ liệu; nhấn mạnh luồng đọc (query) từ \texttt{HangHoa} và \texttt{TonKho}.

\textbf{Đối tượng tham gia}: Người dùng, giao diện tìm kiếm, hệ thống, CSDL.

\textbf{Kịch bản chính}: Nhập điều kiện $\rightarrow$ truy vấn $\rightarrow$ ghép thông tin tồn $\rightarrow$ hiển thị (hoặc thông báo không có dữ liệu).

\input{diagrams/tex/5.2_sequence_uc06}

Giải thích sơ đồ: Đây là luồng \textbf{chỉ đọc}. Actor gửi điều kiện tìm; hệ thống truy vấn CSDL lấy danh sách hàng rồi lấy tồn tương ứng; kết quả trả về giao diện. Không có bước ``Lưu'' thay đổi nghiệp vụ. So với UC13, sequence này ngắn hơn và không có nhánh cập nhật tồn.

\subsection{UC23 -- Lập phiếu nhập kho}

\textbf{Bối cảnh}: Có hai actor theo thời gian: NV lập phiếu trước, QL duyệt sau. Hành vi cập nhật tồn chỉ xảy ra ở nhánh duyệt.

\textbf{Đối tượng tham gia}: NV kho, QL kho, giao diện phiếu nhập, hệ thống, CSDL (\texttt{PhieuNhapKho}, \texttt{ChiTietPhieuNhap}, \texttt{TonKho}, \texttt{LichSuThaoTac}).

\textbf{Kịch bản chính}:
\begin{enumerate}[leftmargin=1.5cm]
    \item NV tạo phiếu, thêm dòng chi tiết, gửi duyệt (trạng thái Chờ duyệt).
    \item QL mở phiếu, chọn duyệt hoặc từ chối.
    \item Nếu duyệt: cập nhật trạng thái, cộng tồn, ghi log.
    \item Nếu từ chối: giữ tồn, lưu lý do.
\end{enumerate}

\input{diagrams/tex/5.3_sequence_uc09}

Giải thích sơ đồ: Sequence có \textbf{hai actor theo thời gian}: NV tạo phiếu và gửi duyệt trước; QL xuất hiện ở bước duyệt. Các thông điệp thể hiện lưu phiếu/chi tiết (trạng thái Chờ duyệt), rồi nhánh duyệt: cập nhật trạng thái, gọi cập nhật \texttt{TonKho} (tăng), ghi log. Nhánh từ chối (nếu có trên sơ đồ/mô tả) không gọi tăng tồn. Sơ đồ gắn hành vi với cấu trúc lớp phiếu--chi tiết--tồn đã nêu ở Ch.4.

\subsection{UC28 -- Lập phiếu xuất kho}

\textbf{Bối cảnh}: Tương tự nhập nhưng có thêm tương tác kiểm tra tồn trước khi chấp nhận dòng hàng; duyệt thì trừ tồn.

\textbf{Đối tượng tham gia}: NV, QL, giao diện phiếu xuất, hệ thống, CSDL phiếu xuất và tồn kho.

\textbf{Kịch bản chính}: Tạo phiếu $\rightarrow$ thêm dòng (kiểm tra tồn) $\rightarrow$ gửi duyệt $\rightarrow$ QL duyệt $\rightarrow$ giảm tồn + log.

\input{diagrams/tex/5.4_sequence_uc10}

Giải thích sơ đồ: Khác UC23 ở chỗ trước khi chấp nhận dòng hàng, hệ thống phải \textbf{hỏi tồn} (\texttt{kiemTraDu}). Chỉ khi đủ, chi tiết mới được lưu. Sau duyệt, thông điệp cập nhật tồn theo hướng \textbf{giảm}. Sequence làm rõ ràng buộc an toàn: không cho xuất vượt số lượng hiện có.

\section{Phân tích hành vi đối tượng phiếu (biểu đồ trạng thái)}

Sequence mô tả tương tác \textbf{giữa} các đối tượng; state machine mô tả hành vi \textbf{của chính} đối tượng phiếu qua các trạng thái.

\subsection{Phiếu nhập kho}

Trạng thái điển hình: Mới tạo $\rightarrow$ Chờ duyệt $\rightarrow$ Đã duyệt / Từ chối; có thể Hủy khi được phép. Chỉ ở ``Đã duyệt'' mới làm thay đổi tồn kho.

\input{diagrams/tex/5.5_state_phieu_nhap}

Giải thích sơ đồ: Hình tròn/oval (hoặc bo tròn) là các \textbf{trạng thái} của một phiếu nhập. Mũi tên là \textbf{sự kiện chuyển trạng thái}: gửi duyệt, duyệt, từ chối, hủy. Ý nghĩa: ở ``Chờ duyệt'' phiếu không được sửa tùy tiện; ``Đã duyệt'' là trạng thái kết thúc nghiệp vụ thành công và là điều kiện để tăng tồn; ``Từ chối'' kết thúc mà không đụng tồn. State bổ sung cho sequence: sequence nói \textit{ai gọi ai}, state nói \textit{phiếu đang ở pha nào}.

\subsection{Phiếu xuất kho}

Cùng mô hình trạng thái với phiếu nhập (Mới tạo, Chờ duyệt, Đã duyệt, Từ chối, Hủy).

\input{diagrams/tex/5.6_state_phieu_xuat}

Giải thích sơ đồ: Máy trạng thái phiếu xuất \textbf{đối xứng} phiếu nhập: cùng các trạng thái và sự kiện chuyển. Khác biệt nghiệp vụ nằm ở hành động kèm theo khi vào ``Đã duyệt'' (giảm tồn thay vì tăng), không nằm ở hình dạng sơ đồ trạng thái.

\section{Kết luận chương}

Chương 5 hoàn thành \textbf{phân tích hành vi} bằng \textbf{biểu đồ trình tự} (và state bổ sung).

Tổng kết ba giai đoạn:
\begin{enumerate}[leftmargin=1.5cm]
    \item \textbf{Chức năng} (Ch.3): phân tích bằng Use Case;
    \item \textbf{Cấu trúc} (Ch.4): phân tích bằng Class;
    \item \textbf{Hành vi} (Ch.5): phân tích bằng Sequence.
\end{enumerate}

Tiếp theo: thiết kế chi tiết (Ch.6) và CSDL/giao diện (Ch.7).
