\section{Đặc tả use case}

\subsection{UC01 -- Đăng nhập}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Tất cả.
    \item \textbf{Mục đích}: Vào hệ thống sau khi xác thực.
    \item \textbf{Điều kiện trước}: Có tài khoản đang hoạt động.
    \item \textbf{Luồng chính}: Mở form $\rightarrow$ nhập tên/mật khẩu $\rightarrow$ kiểm tra $\rightarrow$ tạo phiên, vào trang chủ.
    \item \textbf{Luồng thay thế}: Sai thông tin / bị khóa $\rightarrow$ báo lỗi.
    \item \textbf{Điều kiện sau}: Có phiên làm việc.
\end{itemize}

\subsection{UC02 -- Đăng xuất}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Người đã đăng nhập.
    \item \textbf{Mục đích}: Kết thúc phiên.
    \item \textbf{Luồng chính}: Chọn đăng xuất $\rightarrow$ xóa phiên $\rightarrow$ về form đăng nhập.
\end{itemize}

\subsection{UC03 -- Thêm tài khoản}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Quản trị viên.
    \item \textbf{Mục đích}: Tạo tài khoản mới với vai trò.
    \item \textbf{Luồng chính}: Nhập họ tên, tên đăng nhập, mật khẩu, vai trò $\rightarrow$ kiểm tra trùng $\rightarrow$ mã hóa, lưu.
    \item \textbf{Luồng thay thế}: Tên đăng nhập trùng $\rightarrow$ báo lỗi.
\end{itemize}

\subsection{UC04 -- Thêm nhà cung cấp}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: NV kho, QL kho.
    \item \textbf{Mục đích}: Thêm NCC vào danh mục.
    \item \textbf{Luồng chính}: Nhập mã, tên, địa chỉ, SĐT, email $\rightarrow$ kiểm tra mã $\rightarrow$ lưu.
\end{itemize}

\subsection{UC05 -- Thêm khách hàng}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: NV kho, QL kho.
    \item \textbf{Mục đích}: Thêm khách hàng vào danh mục.
    \item \textbf{Luồng chính}: Tương tự UC04; có thể thêm mã số thuế.
\end{itemize}

\subsection{UC06 -- Thêm hàng hóa}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: NV kho, QL kho.
    \item \textbf{Mục đích}: Thêm mặt hàng mới.
    \item \textbf{Điều kiện trước}: Đã có nhóm hàng, đơn vị tính.
    \item \textbf{Luồng chính}: Nhập mã, tên, nhóm, ĐVT, giá $\rightarrow$ kiểm tra mã $\rightarrow$ lưu + khởi tạo tồn = 0 tại các kho.
\end{itemize}

\subsection{UC07 -- Tìm kiếm hàng hóa}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Người có quyền.
    \item \textbf{Mục đích}: Tra cứu hàng theo mã/tên/nhóm/kho.
    \item \textbf{Luồng chính}: Nhập từ khóa, lọc $\rightarrow$ hiển thị danh sách kèm tồn.
\end{itemize}

\subsection{UC08 -- Thêm kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: QL kho, QTV.
    \item \textbf{Mục đích}: Tạo kho mới.
    \item \textbf{Luồng chính}: Nhập mã, tên, địa chỉ, phụ trách $\rightarrow$ lưu.
\end{itemize}

\subsection{UC09 -- Lập phiếu nhập kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: NV kho (lập), QL kho (duyệt).
    \item \textbf{Mục đích}: Ghi nhận nhập hàng và cập nhật tồn khi duyệt.
    \item \textbf{Luồng chính}: (1) NV chọn NCC, kho, thêm dòng hàng; (2) Gửi duyệt; (3) QL duyệt $\rightarrow$ tăng tồn, ghi log.
    \item \textbf{Luồng thay thế}: QL từ chối $\rightarrow$ tồn không đổi.
\end{itemize}

\subsection{UC10 -- Lập phiếu xuất kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: NV kho (lập), QL kho (duyệt).
    \item \textbf{Mục đích}: Ghi nhận xuất hàng và giảm tồn khi duyệt.
    \item \textbf{Luồng chính}: (1) NV chọn KH, kho, thêm dòng (kiểm tra tồn đủ); (2) Gửi duyệt; (3) QL duyệt $\rightarrow$ giảm tồn.
    \item \textbf{Luồng thay thế}: Tồn không đủ / từ chối $\rightarrow$ không giảm tồn.
\end{itemize}

\section{Kết luận chương}

Chương 3 xác định 3 tác nhân, BFD/DFD và \textbf{10 use case} theo mục tiêu cụ thể (không gộp thành ``quản lý \ldots'' chung). Đặc tả đủ 10 UC làm cơ sở cho sơ đồ lớp và sequence các chương sau.
