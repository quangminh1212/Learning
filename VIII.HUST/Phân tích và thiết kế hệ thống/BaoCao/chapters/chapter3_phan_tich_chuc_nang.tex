\section{Phân tích tác nhân}

\hline

\hline
\end{longtable}

\section{Nguyên tắc xác định use case}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Use case} biểu diễn \textbf{một mục tiêu cụ thể} mà actor muốn đạt được khi dùng hệ thống (ví dụ: ``Thêm hàng hóa'', ``Lập phiếu nhập kho'').
    \item \textbf{Không} coi tên nghiệp vụ chung như ``Quản lý hàng hóa'', ``Quản lý nhà cung cấp'' là \textit{một} use case -- đó là gói chức năng, cần phân rã.
    \item \textbf{Đăng nhập / Đăng xuất} không phải nghiệp vụ kho mà là \textbf{use case hệ thống}, vẫn được đặc tả vì điều kiện để actor thực hiện các use case khác.
    \item Sơ đồ \textbf{tổng quát} thể hiện các nhóm chức năng; sơ đồ \textbf{phân rã} thể hiện từng use case cụ thể.
\end{itemize}

\section{Đặc tả use case}

Mỗi đặc tả gồm: tác nhân, mục tiêu, điều kiện trước, luồng chính, luồng thay thế (nếu có), điều kiện sau.

\subsection{UC01 -- Đăng nhập}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Loại}: Use case hệ thống (hỗ trợ truy cập).
    \item \textbf{Tác nhân}: Tất cả người dùng.
    \item \textbf{Mục tiêu}: Xác thực danh tính và bắt đầu phiên làm việc trên hệ thống.
    \item \textbf{Điều kiện trước}: Người dùng đã có tài khoản ở trạng thái hoạt động.
    \item \textbf{Luồng chính}:
    \begin{enumerate}
        \item Người dùng mở trang đăng nhập.
        \item Nhập tên đăng nhập và mật khẩu, nhấn ``Đăng nhập''.
        \item Hệ thống kiểm tra tài khoản tồn tại, mật khẩu đúng, trạng thái hoạt động.
        \item Hệ thống tạo phiên làm việc, ghi log đăng nhập, chuyển đến trang chủ theo vai trò.
    \end{enumerate}
    \item \textbf{Luồng thay thế}: Sai tên/mật khẩu hoặc tài khoản bị khóa $\rightarrow$ thông báo lỗi, không tạo phiên, cho phép nhập lại.
    \item \textbf{Điều kiện sau}: Phiên làm việc được tạo; người dùng truy cập được các chức năng theo quyền.
\end{itemize}

\subsection{UC02 -- Đăng xuất}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Loại}: Use case hệ thống.
    \item \textbf{Tác nhân}: Người dùng đã đăng nhập.
    \item \textbf{Mục tiêu}: Kết thúc phiên làm việc an toàn.
    \item \textbf{Điều kiện trước}: Đang có phiên hợp lệ.
    \item \textbf{Luồng chính}: Người dùng chọn ``Đăng xuất'' $\rightarrow$ hệ thống xóa phiên, ghi log $\rightarrow$ chuyển về trang đăng nhập.
    \item \textbf{Điều kiện sau}: Phiên bị hủy; không còn truy cập chức năng nội bộ cho đến khi đăng nhập lại.
\end{itemize}

\subsection{UC03 -- Thêm tài khoản}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Quản trị viên.
    \item \textbf{Mục tiêu}: Tạo tài khoản người dùng mới với vai trò xác định.
    \item \textbf{Điều kiện trước}: Quản trị viên đã đăng nhập.
    \item \textbf{Luồng chính}:
    \begin{enumerate}
        \item Chọn chức năng quản lý tài khoản $\rightarrow$ ``Thêm tài khoản''.
        \item Nhập họ tên, tên đăng nhập, mật khẩu, email (nếu có), chọn vai trò (QTV / QL kho / NV kho), trạng thái.
        \item Hệ thống kiểm tra tên đăng nhập chưa tồn tại.
        \item Mã hóa mật khẩu, lưu bản ghi \texttt{NguoiDung}, hiển thị thông báo thành công.
    \end{enumerate}
    \item \textbf{Luồng thay thế}: Tên đăng nhập trùng $\rightarrow$ báo lỗi, yêu cầu đổi tên.
    \item \textbf{Điều kiện sau}: Tài khoản mới tồn tại, có thể đăng nhập (nếu trạng thái hoạt động).
\end{itemize}

\subsection{UC04 -- Thêm nhà cung cấp}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho, Quản lý kho.
    \item \textbf{Mục tiêu}: Thêm nhà cung cấp vào danh mục để phục vụ lập phiếu nhập.
    \item \textbf{Điều kiện trước}: Người dùng đã đăng nhập với quyền phù hợp.
    \item \textbf{Luồng chính}:
    \begin{enumerate}
        \item Chọn quản lý nhà cung cấp $\rightarrow$ ``Thêm mới''.
        \item Nhập mã NCC, tên, địa chỉ, số điện thoại, email, người liên hệ.
        \item Hệ thống kiểm tra mã không trùng $\rightarrow$ lưu \texttt{NhaCungCap}.
    \end{enumerate}
    \item \textbf{Luồng thay thế}: Mã trùng $\rightarrow$ thông báo lỗi.
    \item \textbf{Điều kiện sau}: NCC mới xuất hiện trong danh sách, chọn được khi lập phiếu nhập.
\end{itemize}

\subsection{UC05 -- Thêm khách hàng}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho, Quản lý kho.
    \item \textbf{Mục tiêu}: Thêm khách hàng vào danh mục để phục vụ lập phiếu xuất.
    \item \textbf{Điều kiện trước}: Đã đăng nhập với quyền phù hợp.
    \item \textbf{Luồng chính}: Tương tự UC04; nhập mã KH, tên, địa chỉ, SĐT, email, mã số thuế (nếu có), người liên hệ $\rightarrow$ kiểm tra mã $\rightarrow$ lưu \texttt{KhachHang}.
    \item \textbf{Điều kiện sau}: KH mới dùng được khi lập phiếu xuất.
\end{itemize}

\subsection{UC06 -- Thêm hàng hóa}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho, Quản lý kho.
    \item \textbf{Mục tiêu}: Thêm mặt hàng mới vào danh mục và khởi tạo tồn kho.
    \item \textbf{Điều kiện trước}: Đã có nhóm hàng và đơn vị tính trong hệ thống; người dùng đã đăng nhập.
    \item \textbf{Luồng chính}:
    \begin{enumerate}
        \item Chọn quản lý hàng hóa $\rightarrow$ ``Thêm mới''.
        \item Nhập mã hàng, tên; chọn nhóm hàng, đơn vị tính; nhập giá nhập, giá xuất, mô tả.
        \item Hệ thống kiểm tra mã không trùng.
        \item Lưu \texttt{HangHoa}; tạo các bản ghi \texttt{TonKho} với số lượng = 0 tại mọi kho đang hoạt động.
    \end{enumerate}
    \item \textbf{Luồng thay thế}: Mã trùng hoặc thiếu nhóm/ĐVT $\rightarrow$ báo lỗi.
    \item \textbf{Điều kiện sau}: Hàng mới có trong danh mục; tồn ban đầu = 0 tại các kho.
\end{itemize}

\subsection{UC07 -- Tìm kiếm hàng hóa}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Người dùng có quyền tra cứu.
    \item \textbf{Mục tiêu}: Tìm và xem thông tin hàng hóa kèm tồn kho.
    \item \textbf{Điều kiện trước}: Đã đăng nhập.
    \item \textbf{Luồng chính}:
    \begin{enumerate}
        \item Nhập từ khóa (mã/tên) và/hoặc chọn bộ lọc (nhóm hàng, kho, trạng thái).
        \item Hệ thống truy vấn \texttt{HangHoa} và \texttt{TonKho}.
        \item Hiển thị danh sách kết quả kèm số lượng tồn; cho phép xem chi tiết.
    \end{enumerate}
    \item \textbf{Luồng thay thế}: Không có bản ghi phù hợp $\rightarrow$ thông báo danh sách rỗng.
    \item \textbf{Điều kiện sau}: Người dùng nhận được thông tin cần tra cứu (không làm thay đổi dữ liệu).
\end{itemize}

\subsection{UC08 -- Thêm kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Quản lý kho, Quản trị viên.
    \item \textbf{Mục tiêu}: Tạo kho hàng mới trong hệ thống.
    \item \textbf{Điều kiện trước}: Đã đăng nhập với quyền quản lý kho/QTV.
    \item \textbf{Luồng chính}: Nhập mã kho, tên, địa chỉ, người phụ trách, diện tích $\rightarrow$ kiểm tra mã không trùng $\rightarrow$ lưu \texttt{Kho}.
    \item \textbf{Điều kiện sau}: Kho mới sẵn sàng nhận phiếu nhập và theo dõi tồn.
\end{itemize}

\subsection{UC09 -- Lập phiếu nhập kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho (lập phiếu), Quản lý kho (duyệt phiếu).
    \item \textbf{Mục tiêu}: Ghi nhận hàng nhập từ nhà cung cấp; khi duyệt thì cập nhật tăng tồn kho.
    \item \textbf{Điều kiện trước}: Đã có NCC, kho, hàng hóa; NV và QL đã đăng nhập.
    \item \textbf{Luồng chính}:
    \begin{enumerate}
        \item NV chọn lập phiếu nhập: chọn NCC, kho nhập, ngày nhập.
        \item Thêm dòng chi tiết: chọn hàng, số lượng, đơn giá; hệ thống tính thành tiền và tổng tiền.
        \item Gửi duyệt -- trạng thái phiếu: ``Chờ duyệt''.
        \item QL mở phiếu chờ duyệt, kiểm tra thông tin, chọn \textbf{Duyệt}.
        \item Hệ thống đổi trạng thái ``Đã duyệt''; \textbf{cộng} số lượng vào \texttt{TonKho}; ghi log.
    \end{enumerate}
    \item \textbf{Luồng thay thế}:
    \begin{enumerate}
        \item Dữ liệu phiếu không hợp lệ (thiếu dòng hàng) $\rightarrow$ không cho gửi duyệt.
        \item QL \textbf{Từ chối} $\rightarrow$ trạng thái ``Từ chối'', tồn kho không đổi, lưu lý do.
    \end{enumerate}
    \item \textbf{Điều kiện sau}: Phiếu được lưu; nếu đã duyệt thì tồn kho tăng đúng số lượng nhập.
\end{itemize}

\subsection{UC10 -- Lập phiếu xuất kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho (lập), Quản lý kho (duyệt).
    \item \textbf{Mục tiêu}: Ghi nhận hàng xuất cho khách hàng; khi duyệt thì giảm tồn kho.
    \item \textbf{Điều kiện trước}: Đã có KH, kho, hàng; tồn kho đủ để xuất; NV/QL đã đăng nhập.
    \item \textbf{Luồng chính}:
    \begin{enumerate}
        \item NV chọn lập phiếu xuất: chọn KH, kho xuất, ngày xuất.
        \item Thêm dòng hàng: hệ thống \textbf{kiểm tra tồn đủ} tại kho xuất; nếu đủ thì ghi nhận đơn giá, thành tiền.
        \item Gửi duyệt -- ``Chờ duyệt''.
        \item QL duyệt $\rightarrow$ ``Đã duyệt''; \textbf{trừ} \texttt{TonKho}; ghi log.
    \end{enumerate}
    \item \textbf{Luồng thay thế}: Tồn không đủ $\rightarrow$ không cho thêm dòng; QL từ chối $\rightarrow$ tồn không đổi.
    \item \textbf{Điều kiện sau}: Phiếu được lưu; nếu đã duyệt thì tồn kho giảm đúng số lượng xuất.
\end{itemize}

\subsection{UC11 -- Lập phiếu kiểm kê}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho (lập, nhập số thực tế), Quản lý kho (duyệt điều chỉnh).
    \item \textbf{Mục tiêu}: Đối chiếu tồn thực tế với sổ sách và điều chỉnh chênh lệch khi được duyệt.
    \item \textbf{Điều kiện trước}: Đã có kho và hàng; NV/QL đã đăng nhập.
    \item \textbf{Luồng chính}:
    \begin{enumerate}
        \item NV tạo phiếu kiểm kê, chọn kho, ngày kiểm.
        \item Hệ thống nạp danh sách hàng và \textbf{số lượng sổ} từ \texttt{TonKho}.
        \item NV nhập \textbf{số lượng thực tế}; hệ thống tính chênh lệch = thực tế $-$ sổ.
        \item Gửi duyệt. QL xem chênh lệch và duyệt điều chỉnh.
        \item Hệ thống cập nhật \texttt{TonKho} = số thực tế; trạng thái phiếu ``Đã duyệt''; ghi log.
    \end{enumerate}
    \item \textbf{Luồng thay thế}: QL yêu cầu kiểm lại / từ chối $\rightarrow$ tồn không đổi.
    \item \textbf{Điều kiện sau}: Nếu duyệt, tồn kho khớp số thực tế đã kiểm.
\end{itemize}

\subsection{UC12 -- Xem báo cáo nhập xuất tồn}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Quản lý kho, Quản trị viên.
    \item \textbf{Mục tiêu}: Xem tổng hợp tồn hiện tại hoặc nhập--xuất--tồn theo kỳ phục vụ theo dõi vận hành.
    \item \textbf{Điều kiện trước}: Đã đăng nhập với quyền xem báo cáo; đã có dữ liệu phiếu/tồn.
    \item \textbf{Luồng chính}:
    \begin{enumerate}
        \item Chọn loại báo cáo (tồn hiện tại / NXT theo kỳ).
        \item Nhập điều kiện: khoảng ngày, kho, nhóm hàng hoặc mã hàng (tùy chọn).
        \item Hệ thống tổng hợp từ phiếu nhập, phiếu xuất, tồn kho.
        \item Hiển thị bảng kết quả; tùy chọn xuất Excel.
    \end{enumerate}
    \item \textbf{Điều kiện sau}: Người dùng xem được số liệu; không bắt buộc thay đổi dữ liệu gốc.
\end{itemize}

\section{Kết luận chương}

Chương 3 hoàn thành \textbf{phân tích chức năng}: xác định tác nhân, xây dựng use case tổng quát và phân rã, đặc tả đủ 12 use case (mục tiêu, luồng chính/thay thế, điều kiện). Đây là đầu vào cho \textbf{phân tích cấu trúc} (biểu đồ lớp -- Ch.4) và \textbf{phân tích hành vi} (biểu đồ trình tự -- Ch.5).
