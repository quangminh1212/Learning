\section{Đặc tả use case}

\subsection{UC01 -- Đăng nhập}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Vào hệ thống sau xác thực.
    \item \textbf{Trước}: Có tài khoản hoạt động.
    \item \textbf{Chính}: Nhập tên/MK $\rightarrow$ kiểm tra $\rightarrow$ tạo phiên.
    \item \textbf{Thay thế}: Sai TT / khóa $\rightarrow$ lỗi.
\end{itemize}

\subsection{UC02 -- Đăng xuất}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Kết thúc phiên. Chọn đăng xuất $\rightarrow$ xóa phiên.
\end{itemize}

\subsection{UC03 -- Thêm tài khoản}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Actor}: QTV. \textbf{Mục tiêu}: Tạo tài khoản mới.
    \item \textbf{Chính}: Nhập họ tên, tên ĐN, MK, vai trò $\rightarrow$ kiểm tra trùng $\rightarrow$ lưu.
\end{itemize}

\subsection{UC04 -- Thêm nhà cung cấp}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Thêm NCC. Nhập mã, tên, liên hệ $\rightarrow$ lưu.
\end{itemize}

\subsection{UC05 -- Thêm khách hàng}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Thêm KH. Tương tự UC04 (+ MST nếu có).
\end{itemize}

\subsection{UC06 -- Thêm hàng hóa}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Thêm mặt hàng. Nhập mã, tên, nhóm, ĐVT, giá $\rightarrow$ lưu + tồn = 0 các kho.
\end{itemize}

\subsection{UC07 -- Tìm kiếm hàng hóa}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Tra cứu hàng theo mã/tên/nhóm, kèm tồn.
\end{itemize}

\subsection{UC08 -- Thêm kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Tạo kho mới (mã, tên, địa chỉ, phụ trách).
\end{itemize}

\subsection{UC09 -- Lập phiếu nhập kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Actor}: NV lập, QL duyệt.
    \item \textbf{Mục tiêu}: Ghi nhận nhập hàng và cập nhật tồn khi duyệt.
    \item \textbf{Chính}: NV chọn NCC, kho, thêm dòng $\rightarrow$ gửi duyệt $\rightarrow$ QL duyệt $\rightarrow$ tăng tồn.
    \item \textbf{Thay thế}: Từ chối $\rightarrow$ tồn không đổi.
\end{itemize}

\subsection{UC10 -- Lập phiếu xuất kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Actor}: NV lập, QL duyệt.
    \item \textbf{Mục tiêu}: Ghi nhận xuất hàng và giảm tồn khi duyệt.
    \item \textbf{Chính}: NV chọn KH, kho, thêm dòng (tồn đủ) $\rightarrow$ gửi duyệt $\rightarrow$ QL duyệt $\rightarrow$ giảm tồn.
\end{itemize}
