\section{Giới thiệu giai đoạn}

\textbf{Phân tích chức năng} trả lời: hệ thống cung cấp \textbf{chức năng gì}, \textbf{ai} sử dụng. Biểu đồ đặc trưng: \textbf{Use Case Diagram} (tổng quát, phân rã) và \textbf{đặc tả use case}.

Hai giai đoạn sau:
\begin{itemize}[leftmargin=1.5cm]
    \item Ch.4 \textbf{Phân tích cấu trúc} -- biểu đồ lớp;
    \item Ch.5 \textbf{Phân tích hành vi} -- biểu đồ trình tự.
\end{itemize}

\section{Nguyên tắc use case}

\begin{itemize}[leftmargin=1.5cm]
    \item Use case = \textbf{một mục tiêu cụ thể} của actor (vd.\ Thêm hàng hóa).
    \item Nghiệp vụ ``Quản lý \ldots'' \textbf{không} phải một use case.
    \item Đăng nhập/đăng xuất: use case \textbf{hệ thống}, không phải nghiệp vụ kho.
\end{itemize}

\section{Đặc tả use case}

\subsection{UC01 -- Đăng nhập}
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Loại}: Use case hệ thống.
    \item \textbf{Mục tiêu}: Vào hệ thống sau xác thực.
    \item \textbf{Chính}: Nhập tên/MK $\rightarrow$ kiểm tra $\rightarrow$ tạo phiên.
    \item \textbf{Thay thế}: Sai TT / khóa $\rightarrow$ lỗi.
\end{itemize}

\subsection{UC02 -- Đăng xuất}
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Loại}: Use case hệ thống.
    \item \textbf{Mục tiêu}: Kết thúc phiên.
\end{itemize}

\subsection{UC03 -- Thêm tài khoản}
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Actor}: QTV. \textbf{Mục tiêu}: Tạo tài khoản mới.
    \item \textbf{Chính}: Nhập TT + vai trò $\rightarrow$ kiểm tra trùng $\rightarrow$ lưu.
\end{itemize}

\subsection{UC04 -- Thêm nhà cung cấp}
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Thêm NCC vào danh mục.
\end{itemize}

\subsection{UC05 -- Thêm khách hàng}
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Thêm KH vào danh mục.
\end{itemize}

\subsection{UC06 -- Thêm hàng hóa}
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Thêm mặt hàng; khởi tạo tồn = 0 tại các kho.
\end{itemize}

\subsection{UC07 -- Tìm kiếm hàng hóa}
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Tra cứu hàng kèm tồn kho.
\end{itemize}

\subsection{UC08 -- Thêm kho}
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Tạo kho mới.
\end{itemize}

\subsection{UC09 -- Lập phiếu nhập kho}
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Actor}: NV lập, QL duyệt.
    \item \textbf{Mục tiêu}: Ghi nhận nhập; duyệt thì tăng tồn.
\end{itemize}

\subsection{UC10 -- Lập phiếu xuất kho}
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Actor}: NV lập, QL duyệt.
    \item \textbf{Mục tiêu}: Ghi nhận xuất (kiểm tra tồn); duyệt thì giảm tồn.
\end{itemize}

\subsection{UC11 -- Lập phiếu kiểm kê}
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Đối chiếu thực tế -- sổ sách; duyệt thì điều chỉnh tồn.
\end{itemize}

\subsection{UC12 -- Xem báo cáo nhập xuất tồn}
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Xem tổng hợp tồn / NXT theo điều kiện.
\end{itemize}
