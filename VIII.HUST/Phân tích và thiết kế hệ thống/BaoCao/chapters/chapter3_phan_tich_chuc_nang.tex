\section{Nguyên tắc tách use case}

\begin{itemize}[leftmargin=1.5cm]
    \item Mỗi use case là \textbf{một mục tiêu cụ thể} mà actor muốn đạt (ví dụ: ``Thêm hàng hóa'').
    \item Tên nghiệp vụ kiểu ``Quản lý hàng hóa'' là gói chung, \textbf{không} đặt làm một use case duy nhất.
    \item \textbf{Đăng nhập / Đăng xuất} là use case \textbf{hệ thống} (hỗ trợ truy cập), không phải nghiệp vụ quản lý kho, nhưng vẫn cần đặc tả vì là điểm vào của mọi actor.
\end{itemize}

\section{Đặc tả use case}

Phần đặc tả mô tả chi tiết từng use case: bối cảnh, điều kiện, các bước tương tác giữa actor và hệ thống, các nhánh thay thế và kết quả sau khi kết thúc. Đây là cơ sở để xây dựng biểu đồ lớp (cấu trúc) và biểu đồ trình tự (hành vi) ở các chương sau.

\subsection{UC01 -- Đăng nhập}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên use case}: Đăng nhập.
    \item \textbf{Mã}: UC01.
    \item \textbf{Loại}: Use case hệ thống (hỗ trợ truy cập an toàn).
    \item \textbf{Tác nhân chính}: Tất cả người dùng (QTV, QL kho, NV kho).
    \item \textbf{Mô tả tóm tắt}: Người dùng cung cấp tên đăng nhập và mật khẩu; hệ thống xác thực và mở phiên làm việc phù hợp với vai trò.
    \item \textbf{Mục tiêu}: Được hệ thống công nhận danh tính và vào được khu vực chức năng theo quyền.
    \item \textbf{Điều kiện trước}:
    \begin{enumerate}
        \item Người dùng đã được cấp tài khoản trên hệ thống.
        \item Tài khoản đang ở trạng thái hoạt động (không bị khóa).
        \item Người dùng đang ở màn hình đăng nhập (chưa có phiên).
    \end{enumerate}
    \item \textbf{Điều kiện sau (thành công)}:
    \begin{enumerate}
        \item Phiên làm việc được tạo và gắn với người dùng.
        \item Hệ thống ghi nhận thời điểm đăng nhập (log).
        \item Người dùng được chuyển tới trang chủ / giao diện theo vai trò.
    \end{enumerate}
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item Người dùng mở trang đăng nhập của hệ thống.
        \item Hệ thống hiển thị form gồm ô tên đăng nhập, mật khẩu và nút ``Đăng nhập''.
        \item Người dùng nhập tên đăng nhập và mật khẩu, nhấn ``Đăng nhập''.
        \item Hệ thống nhận dữ liệu, tìm tài khoản theo tên đăng nhập trong cơ sở dữ liệu.
        \item Hệ thống so khớp mật khẩu (sau khi mã hóa / kiểm tra hash) và kiểm tra trạng thái tài khoản.
        \item Nếu hợp lệ, hệ thống tạo phiên (session/token), lưu thông tin người dùng và vai trò vào phiên.
        \item Hệ thống ghi log đăng nhập thành công.
        \item Hệ thống chuyển hướng sang trang chủ tương ứng vai trò (QTV / QL / NV).
    \end{enumerate}
    \item \textbf{Luồng thay thế / ngoại lệ}:
    \begin{enumerate}
        \item \textit{A1 -- Sai tên đăng nhập hoặc mật khẩu}: Ở bước 5, thông tin không khớp. Hệ thống báo ``Sai tên đăng nhập hoặc mật khẩu'', giữ nguyên form, cho phép nhập lại. Không tạo phiên.
        \item \textit{A2 -- Tài khoản bị khóa}: Tài khoản tồn tại nhưng trạng thái không hoạt động. Hệ thống báo ``Tài khoản đã bị khóa, liên hệ quản trị viên''. Không tạo phiên.
        \item \textit{A3 -- Bỏ trống trường bắt buộc}: Hệ thống báo yêu cầu nhập đủ tên đăng nhập và mật khẩu, không gửi kiểm tra CSDL.
    \end{enumerate}
    \item \textbf{Yêu cầu đặc biệt}: Mật khẩu không hiển thị dạng rõ trên giao diện; truyền tải an toàn theo chính sách triển khai.
\end{itemize}

\subsection{UC02 -- Đăng xuất}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên use case}: Đăng xuất.
    \item \textbf{Mã}: UC02.
    \item \textbf{Loại}: Use case hệ thống.
    \item \textbf{Tác nhân chính}: Người dùng đang có phiên hợp lệ.
    \item \textbf{Mô tả tóm tắt}: Người dùng chủ động kết thúc phiên để không cho người khác tiếp tục sử dụng trên cùng trình duyệt/máy.
    \item \textbf{Mục tiêu}: Hủy phiên làm việc hiện tại một cách an toàn.
    \item \textbf{Điều kiện trước}: Người dùng đã đăng nhập thành công (có phiên).
    \item \textbf{Điều kiện sau}: Phiên bị xóa; mọi yêu cầu chức năng nội bộ sau đó đều yêu cầu đăng nhập lại.
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item Từ giao diện hệ thống, người dùng chọn ``Đăng xuất''.
        \item Hệ thống xác nhận thao tác (nếu có hộp thoại xác nhận).
        \item Hệ thống xóa / vô hiệu hóa phiên hiện tại.
        \item Hệ thống ghi log đăng xuất.
        \item Hệ thống chuyển về trang đăng nhập.
    \end{enumerate}
    \item \textbf{Luồng thay thế}: Người dùng đóng trình duyệt đột ngột -- phiên hết hạn theo cơ chế timeout (xử lý phía máy chủ).
\end{itemize}

\subsection{UC03 -- Thêm tài khoản}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên use case}: Thêm tài khoản.
    \item \textbf{Mã}: UC03.
    \item \textbf{Tác nhân chính}: Quản trị viên.
    \item \textbf{Mô tả tóm tắt}: Quản trị viên tạo tài khoản mới cho nhân sự (QL kho, NV kho hoặc QTV khác), gán vai trò để phân quyền sử dụng hệ thống.
    \item \textbf{Mục tiêu}: Có thêm một tài khoản hợp lệ trong hệ thống, sẵn sàng đăng nhập.
    \item \textbf{Điều kiện trước}:
    \begin{enumerate}
        \item Quản trị viên đã đăng nhập (UC01) với vai trò QTV.
        \item Có thông tin nhân sự cần cấp tài khoản (họ tên, vai trò dự kiến).
    \end{enumerate}
    \item \textbf{Điều kiện sau}: Bản ghi \texttt{NguoiDung} mới được lưu; mật khẩu đã mã hóa; tài khoản có thể đăng nhập nếu trạng thái ``Hoạt động''.
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item QTV mở ``Quản lý tài khoản'' (có thể từ UC07 xem danh sách).
        \item QTV chọn ``Thêm tài khoản''.
        \item Hệ thống mở form: họ tên, tên đăng nhập, mật khẩu, nhập lại mật khẩu, email (nếu có), vai trò, trạng thái.
        \item QTV nhập đủ thông tin và nhấn ``Lưu''.
        \item Hệ thống kiểm tra trường bắt buộc, định dạng, và tên đăng nhập chưa tồn tại.
        \item Hệ thống mã hóa mật khẩu, lưu tài khoản, ghi log (tuỳ chọn).
        \item Thông báo thành công, làm mới danh sách.
    \end{enumerate}
    \item \textbf{Luồng thay thế}:
    \begin{enumerate}
        \item \textit{A1 -- Tên đăng nhập trùng}: Báo lỗi, yêu cầu đổi tên, giữ các trường khác.
        \item \textit{A2 -- Mật khẩu nhập lại không khớp}: Báo lỗi, không lưu.
        \item \textit{A3 -- Thiếu trường bắt buộc}: Đánh dấu trường thiếu, không gửi lưu.
    \end{enumerate}
\end{itemize}

\subsection{UC04 -- Sửa tài khoản}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên use case}: Sửa tài khoản.
    \item \textbf{Mã}: UC04.
    \item \textbf{Tác nhân chính}: Quản trị viên.
    \item \textbf{Mô tả tóm tắt}: Cập nhật thông tin tài khoản đã có (họ tên, email, vai trò, trạng thái; đổi mật khẩu nếu cần) khi nhân sự đổi vị trí hoặc cần hiệu chỉnh.
    \item \textbf{Mục tiêu}: Bản ghi \texttt{NguoiDung} phản ánh đúng thông tin và quyền mới nhất.
    \item \textbf{Điều kiện trước}: QTV đã đăng nhập; đã tồn tại tài khoản cần sửa (thường chọn từ UC07 hoặc kết quả UC06).
    \item \textbf{Điều kiện sau}: Thông tin tài khoản được cập nhật; tên đăng nhập thường \textbf{không} đổi (khóa định danh).
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item QTV tìm/chọn tài khoản cần sửa trên danh sách.
        \item Chọn ``Sửa''. Hệ thống nạp form với dữ liệu hiện tại.
        \item QTV chỉnh họ tên, email, vai trò, trạng thái; nếu đổi mật khẩu thì nhập mật khẩu mới + xác nhận.
        \item Nhấn ``Lưu''. Hệ thống validate; nếu có mật khẩu mới thì mã hóa lại.
        \item Thông báo thành công, hiển thị lại danh sách/chi tiết.
    \end{enumerate}
    \item \textbf{Luồng thay thế}:
    \begin{enumerate}
        \item \textit{A1 -- Hủy sửa}: Quay lại danh sách, không lưu.
        \item \textit{A2 -- Mật khẩu xác nhận không khớp}: Báo lỗi, không lưu.

\subsection{UC05 -- Xóa tài khoản}

    \item \textbf{Mã}: UC05.
    \item \textbf{Tác nhân chính}: Quản trị viên.
    \item \textbf{Mô tả tóm tắt}: Gỡ tài khoản khỏi hệ thống hoặc khóa vĩnh viễn. Nếu tài khoản đã lập phiếu / có log thì không xóa cứng để giữ toàn vẹn lịch sử.
    \item \textbf{Mục tiêu}: Tài khoản không còn đăng nhập được (đã xóa hoặc đã khóa).
    \item \textbf{Điều kiện trước}: QTV đã đăng nhập; đã chọn một tài khoản (không phải tài khoản đang đăng nhập của chính QTV -- khuyến nghị).
    \item \textbf{Điều kiện sau}: Tài khoản bị xóa \textbf{hoặc} chuyển trạng thái ``Khóa'' / ``Ngừng''.
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item QTV chọn tài khoản $\rightarrow$ ``Xóa''.
        \item Hệ thống kiểm tra có phiếu / lịch sử thao tác tham chiếu người dùng không.
        \item \textbf{Nếu chưa có dữ liệu liên quan}: yêu cầu xác nhận $\rightarrow$ xóa bản ghi $\rightarrow$ thông báo thành công.
        \item \textbf{Nếu đã có phiếu/log}: thông báo không xóa cứng; đề xuất ``Khóa tài khoản''.
        \item QTV xác nhận khóa $\rightarrow$ cập nhật trạng thái; người dùng không đăng nhập được nữa (UC01 từ chối).
    \end{enumerate}
    \item \textbf{Luồng thay thế}: Hủy xác nhận $\rightarrow$ không thay đổi; cố xóa chính mình $\rightarrow$ hệ thống chặn.
