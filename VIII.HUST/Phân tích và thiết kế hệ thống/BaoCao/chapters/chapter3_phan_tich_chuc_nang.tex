\section{Nguyên tắc use case}

\begin{itemize}[leftmargin=1.5cm]
    \item Use case = \textbf{một mục tiêu cụ thể} actor muốn đạt.
    \item Nghiệp vụ ``Quản lý \ldots'' \textbf{không} phải một use case.
    \item \textbf{Tổng quát}: vẽ nhóm chức năng (từ nghiệp vụ).
    \item \textbf{Phân rã}: vẽ từng hành động (Thêm, Sửa, Xóa, Tìm\ldots).
\end{itemize}

\section{Đặc tả use case}

\subsection{UC01 -- Đăng nhập}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Actor}: Tất cả. \textbf{Mục tiêu}: Vào hệ thống.
    \item \textbf{Trước}: Có TK đang hoạt động.
    \item \textbf{Chính}: Form đăng nhập $\rightarrow$ nhập tên/MK $\rightarrow$ kiểm tra $\rightarrow$ tạo phiên.
    \item \textbf{Thay thế}: Sai TT / bị khóa $\rightarrow$ báo lỗi.
\end{itemize}

\subsection{UC02 -- Đăng xuất}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Kết thúc phiên. Chọn đăng xuất $\rightarrow$ xóa phiên.
\end{itemize}

\subsection{UC03 -- Thêm tài khoản}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Actor}: QTV. \textbf{Mục tiêu}: Tạo TK mới.
    \item \textbf{Chính}: Nhập họ tên, tên ĐN, MK, vai trò $\rightarrow$ kiểm tra trùng $\rightarrow$ lưu.
\end{itemize}

\subsection{UC04 -- Sửa tài khoản}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Cập nhật TT/vai trò TK đã có. Chọn TK $\rightarrow$ sửa $\rightarrow$ lưu.
\end{itemize}

\subsection{UC05 -- Xóa/khóa tài khoản}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Khóa hoặc xóa TK. Chọn TK $\rightarrow$ xác nhận.
\end{itemize}

\subsection{UC06 -- Tìm kiếm tài khoản}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Mục tiêu}: Xem/tìm danh sách TK theo điều kiện.
\end{itemize}

\subsection{UC07--UC10 -- CRUD nhà cung cấp}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{UC07 Thêm}: nhập mã, tên, liên hệ $\rightarrow$ lưu.
    \item \textbf{UC08 Sửa}: chọn $\rightarrow$ sửa $\rightarrow$ lưu.
    \item \textbf{UC09 Xóa}: xóa/ngừng nếu còn phiếu liên quan.
    \item \textbf{UC10 Tìm}: theo mã/tên $\rightarrow$ danh sách.
\end{itemize}

\subsection{UC11--UC14 -- CRUD khách hàng}

Tương tự UC07--UC10, đối tượng khách hàng.

\subsection{UC15--UC18 -- CRUD hàng hóa}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{UC15 Thêm}: mã, tên, nhóm, ĐVT, giá $\rightarrow$ lưu + tồn = 0.
    \item \textbf{UC16 Sửa}; \textbf{UC17 Ngừng}; \textbf{UC18 Tìm} (kèm tồn).
\end{itemize}

\subsection{UC19--UC21 -- Quản lý kho}

\textbf{UC19 Thêm kho}; \textbf{UC20 Sửa kho}; \textbf{UC21 Tìm kho}.

\subsection{UC22 -- Lập phiếu nhập kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Actor}: NV. \textbf{Mục tiêu}: Tạo phiếu nhập, gửi duyệt.
    \item \textbf{Chính}: Chọn NCC, kho $\rightarrow$ thêm dòng hàng $\rightarrow$ gửi duyệt.
\end{itemize}

\subsection{UC23 -- Duyệt phiếu nhập kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Actor}: QL. \textbf{Mục tiêu}: Duyệt/từ chối; duyệt $\rightarrow$ tăng tồn.
\end{itemize}

\subsection{UC24 -- Lập phiếu xuất kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Actor}: NV. \textbf{Mục tiêu}: Tạo phiếu xuất (kiểm tra tồn), gửi duyệt.
\end{itemize}

\subsection{UC25 -- Duyệt phiếu xuất kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Actor}: QL. \textbf{Mục tiêu}: Duyệt/từ chối; duyệt $\rightarrow$ giảm tồn.
\end{itemize}
