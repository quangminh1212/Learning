\section{Sơ đồ use case phân rã}

Các use case chính được phân rã thành các nhóm chức năng cụ thể:

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Quản lý người dùng}: UC01 Quản lý tài khoản, UC02 Đăng nhập/đăng xuất.
    \item \textbf{Quản lý đối tác}: UC03 Quản lý nhà cung cấp, UC04 Quản lý khách hàng.
    \item \textbf{Quản lý hàng hóa và kho}: UC05 Quản lý hàng hóa, UC06 Tìm kiếm hàng hóa, UC07 Quản lý kho, UC08 Tìm kiếm kho.
    \item \textbf{Quản lý nhập xuất}: UC09 Lập phiếu nhập kho, UC10 Lập phiếu xuất kho.
    \item \textbf{Quản lý đơn hàng và tài chính}: UC11 Quản lý đơn đặt hàng, UC12 Quản lý thanh toán.
    \item \textbf{Báo cáo}: UC13 Xem báo cáo thống kê.
\end{itemize}

\section{Đặc tả use case}

\subsection{UC01 -- Quản lý tài khoản người dùng}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Quản trị viên.
    \item \textbf{Mục đích}: Tạo, chỉnh sửa, khóa/mở tài khoản và phân quyền.
    \item \textbf{Luồng chính}: Quản trị viên chọn chức năng quản lý tài khoản $\rightarrow$ nhập thông tin tài khoản $\rightarrow$ chọn vai trò $\rightarrow$ lưu.
\end{itemize}

\subsection{UC02 -- Đăng nhập / Đăng xuất}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Tất cả người dùng.
    \item \textbf{Mục đích}: Xác thực người dùng trước khi truy cập hệ thống.
    \item \textbf{Luồng chính}: Người dùng nhập tên đăng nhập, mật khẩu $\rightarrow$ hệ thống kiểm tra $\rightarrow$ chuyển đến trang chủ.
    \item \textbf{Luồng thay thế}: Thông tin sai $\rightarrow$ thông báo lỗi.
\end{itemize}

\subsection{UC03 -- Quản lý nhà cung cấp}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho, Quản lý kho.
    \item \textbf{Mục đích}: Thêm, sửa, xóa, tìm kiếm nhà cung cấp.
    \item \textbf{Luồng chính}: Chọn chức năng quản lý nhà cung cấp $\rightarrow$ nhập thông tin $\rightarrow$ lưu.
\end{itemize}

\subsection{UC04 -- Quản lý khách hàng}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho, Quản lý kho.
    \item \textbf{Mục đích}: Thêm, sửa, xóa, tìm kiếm khách hàng.
    \item \textbf{Luồng chính}: Tương tự UC03.
\end{itemize}

\subsection{UC05 -- Quản lý hàng hóa}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho, Quản lý kho.
    \item \textbf{Mục đích}: Thêm, cập nhật, tìm kiếm, phân loại hàng hóa.
    \item \textbf{Luồng chính}: Chọn chức năng quản lý hàng hóa $\rightarrow$ nhập mã, tên, nhóm hàng, đơn vị tính, giá $\rightarrow$ lưu.
\end{itemize}

\subsection{UC06 -- Tìm kiếm hàng hóa}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Tất cả người dùng có quyền.
    \item \textbf{Mục đích}: Tra cứu hàng hóa theo mã, tên, nhóm hàng hoặc kho.
    \item \textbf{Luồng chính}: Nhập từ khóa $\rightarrow$ hệ thống tìm kiếm $\rightarrow$ hiển thị kết quả kèm tồn kho.
\end{itemize}

\subsection{UC07 -- Quản lý kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Quản lý kho, Quản trị viên.
    \item \textbf{Mục đích}: Tạo, cập nhật, tìm kiếm kho.
    \item \textbf{Luồng chính}: Chọn chức năng quản lý kho $\rightarrow$ nhập mã, tên, địa chỉ, người phụ trách $\rightarrow$ lưu.
\end{itemize}

\subsection{UC08 -- Tìm kiếm kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Tất cả người dùng có quyền.
    \item \textbf{Mục đích}: Tra cứu thông tin kho và tồn kho trong kho.
    \item \textbf{Luồng chính}: Nhập mã/tên kho $\rightarrow$ hệ thống hiển thị thông tin và danh sách hàng tồn.
\end{itemize}

\subsection{UC09 -- Lập phiếu nhập kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho, Quản lý kho.
    \item \textbf{Mục đích}: Ghi nhận hàng hóa nhập vào kho.
    \item \textbf{Luồng chính}: Chọn lập phiếu nhập $\rightarrow$ chọn nhà cung cấp, kho, ngày nhập $\rightarrow$ thêm chi tiết hàng hóa $\rightarrow$ gửi duyệt $\rightarrow$ cập nhật tồn kho.
    \item \textbf{Luồng thay thế}: Tồn kho không đủ chỗ hoặc thông tin sai $\rightarrow$ thông báo lỗi.
\end{itemize}

\subsection{UC10 -- Lập phiếu xuất kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Nhân viên kho, Quản lý kho.
    \item \textbf{Mục đích}: Ghi nhận hàng hóa xuất ra khỏi kho.
    \item \textbf{Luồng chính}: Chọn lập phiếu xuất $\rightarrow$ chọn khách hàng, kho, ngày xuất $\rightarrow$ thêm chi tiết hàng hóa $\rightarrow$ kiểm tra tồn kho $\rightarrow$ gửi duyệt $\rightarrow$ giảm tồn kho.
    \item \textbf{Luồng thay thế}: Tồn kho không đủ $\rightarrow$ thông báo không đủ hàng.
\end{itemize}

\subsection{UC11 -- Quản lý đơn đặt hàng}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Khách hàng, Nhân viên kho.
    \item \textbf{Mục đích}: Tạo, duyệt, theo dõi đơn đặt hàng.
    \item \textbf{Luồng chính}: Tạo đơn hàng $\rightarrow$ chọn khách hàng, hàng hóa, số lượng $\rightarrow$ gửi duyệt $\rightarrow$ tạo phiếu xuất kho khi duyệt.
\end{itemize}

\subsection{UC12 -- Quản lý thanh toán}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Quản lý kho, Quản trị viên.
    \item \textbf{Mục đích}: Ghi nhận, cập nhật chi phí/thanh toán liên quan đến phiếu nhập/xuất.
    \item \textbf{Luồng chính}: Chọn phiếu nhập/xuất $\rightarrow$ nhập số tiền thanh toán, ngày thanh toán, phương thức $\rightarrow$ lưu.
\end{itemize}

\subsection{UC13 -- Xem báo cáo thống kê}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tác nhân}: Quản lý kho, Quản trị viên.
    \item \textbf{Mục đích}: Xem các báo cáo nhập xuất tồn, hàng bán chạy, tồn lâu, giá trị tồn kho.
    \item \textbf{Luồng chính}: Chọn loại báo cáo và khoảng thời gian $\rightarrow$ hệ thống truy vấn dữ liệu $\rightarrow$ hiển thị báo cáo.
\end{itemize}
