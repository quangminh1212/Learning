\section{Nguyên tắc tách use case}

\begin{itemize}[leftmargin=1.5cm]
    \item Mỗi use case là \textbf{một mục tiêu cụ thể} mà actor muốn đạt (ví dụ: ``Thêm hàng hóa'').
    \item Tên nghiệp vụ kiểu ``Quản lý hàng hóa'' là gói chung, \textbf{không} đặt làm một use case duy nhất.
    \item \textbf{Đăng nhập / Đăng xuất} là use case \textbf{hệ thống} (hỗ trợ truy cập), không phải nghiệp vụ quản lý kho, nhưng vẫn cần đặc tả vì là điểm vào của mọi actor.
\end{itemize}

Sơ đồ tổng quát thể hiện các \textbf{nhóm chức năng} và quan hệ với tác nhân. Mỗi oval mức tổng quát là nhóm sẽ được phân rã ở mục sau.

\input{diagrams/tex/3.4_usecase}

Giải thích sơ đồ:
\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Hộp nét đứt}: ranh giới hệ thống quản lý kho -- mọi use case nằm trong hệ thống.
    \item \textbf{Ba actor} (hình người): Quản trị viên (trái), Quản lý kho (phải), Nhân viên kho (dưới) -- vai trò bên ngoài chủ động gọi chức năng.
    \item \textbf{Các oval}: nhóm chức năng mức tổng quát (Đăng nhập/Đăng xuất, Quản lý tài khoản, NCC, KH, hàng hóa, kho, nhập, xuất, kiểm kê, báo cáo tồn). Đây \textbf{chưa} phải use case chi tiết.

    \item \textbf{Ý nghĩa phân tích}: xác định \textit{ai dùng gì} ở mức toàn hệ thống trước khi đi sâu từng mục tiêu.
\end{itemize}

\section{Đặc tả use case}

Phần đặc tả mô tả chi tiết từng use case: bối cảnh, điều kiện, các bước tương tác giữa actor và hệ thống, các nhánh thay thế và kết quả sau khi kết thúc. Đây là cơ sở để xây dựng biểu đồ lớp (cấu trúc) và biểu đồ trình tự (hành vi) ở các chương sau.

\subsection{UC01 -- Đăng nhập}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên use case}: Đăng nhập.
    \item \textbf{Mã}: UC01.
    \item \textbf{Loại}: Use case hệ thống (hỗ trợ truy cập an toàn).
    \item \textbf{Tác nhân chính}: Tất cả người dùng (QTV, QL kho, NV kho).
    \item \textbf{Mô tả tóm tắt}: Người dùng cung cấp tên đăng nhập và mật khẩu; hệ thống xác thực và mở phiên làm việc phù hợp với vai trò.
    \item \textbf{Mục tiêu}: Được hệ thống công nhận danh tính và vào được khu vực chức năng theo quyền.
    \item \textbf{Điều kiện trước}:
    \begin{enumerate}
        \item Người dùng đã được cấp tài khoản trên hệ thống.
        \item Tài khoản đang ở trạng thái hoạt động (không bị khóa).
        \item Người dùng đang ở màn hình đăng nhập (chưa có phiên).
    \end{enumerate}
    \item \textbf{Điều kiện sau (thành công)}:
    \begin{enumerate}
        \item Phiên làm việc được tạo và gắn với người dùng.
        \item Hệ thống ghi nhận thời điểm đăng nhập (log).
        \item Người dùng được chuyển tới trang chủ / giao diện theo vai trò.
    \end{enumerate}
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item Người dùng mở trang đăng nhập của hệ thống.
        \item Hệ thống hiển thị form gồm ô tên đăng nhập, mật khẩu và nút ``Đăng nhập''.
        \item Người dùng nhập tên đăng nhập và mật khẩu, nhấn ``Đăng nhập''.
        \item Hệ thống nhận dữ liệu, tìm tài khoản theo tên đăng nhập trong cơ sở dữ liệu.
        \item Hệ thống so khớp mật khẩu (sau khi mã hóa / kiểm tra hash) và kiểm tra trạng thái tài khoản.
        \item Nếu hợp lệ, hệ thống tạo phiên (session/token), lưu thông tin người dùng và vai trò vào phiên.
        \item Hệ thống ghi log đăng nhập thành công.
        \item Hệ thống chuyển hướng sang trang chủ tương ứng vai trò (QTV / QL / NV).
    \end{enumerate}
    \item \textbf{Luồng thay thế / ngoại lệ}:
    \begin{enumerate}
        \item \textit{A1 -- Sai tên đăng nhập hoặc mật khẩu}: Ở bước 5, thông tin không khớp. Hệ thống báo ``Sai tên đăng nhập hoặc mật khẩu'', giữ nguyên form, cho phép nhập lại. Không tạo phiên.
        \item \textit{A2 -- Tài khoản bị khóa}: Tài khoản tồn tại nhưng trạng thái không hoạt động. Hệ thống báo ``Tài khoản đã bị khóa, liên hệ quản trị viên''. Không tạo phiên.
        \item \textit{A3 -- Bỏ trống trường bắt buộc}: Hệ thống báo yêu cầu nhập đủ tên đăng nhập và mật khẩu, không gửi kiểm tra CSDL.
    \end{enumerate}
    \item \textbf{Yêu cầu đặc biệt}: Mật khẩu không hiển thị dạng rõ trên giao diện; truyền tải an toàn theo chính sách triển khai.
\end{itemize}

\subsection{UC02 -- Đăng xuất}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên use case}: Đăng xuất.
    \item \textbf{Mã}: UC02.
    \item \textbf{Loại}: Use case hệ thống.
    \item \textbf{Tác nhân chính}: Người dùng đang có phiên hợp lệ.
    \item \textbf{Mô tả tóm tắt}: Người dùng chủ động kết thúc phiên để không cho người khác tiếp tục sử dụng trên cùng trình duyệt/máy.
    \item \textbf{Mục tiêu}: Hủy phiên làm việc hiện tại một cách an toàn.
    \item \textbf{Điều kiện trước}: Người dùng đã đăng nhập thành công (có phiên).
    \item \textbf{Điều kiện sau}: Phiên bị xóa; mọi yêu cầu chức năng nội bộ sau đó đều yêu cầu đăng nhập lại.
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item Từ giao diện hệ thống, người dùng chọn ``Đăng xuất''.
        \item Hệ thống xác nhận thao tác (nếu có hộp thoại xác nhận).
        \item Hệ thống xóa / vô hiệu hóa phiên hiện tại.
        \item Hệ thống ghi log đăng xuất.
        \item Hệ thống chuyển về trang đăng nhập.
    \end{enumerate}
    \item \textbf{Luồng thay thế}: Người dùng đóng trình duyệt đột ngột -- phiên hết hạn theo cơ chế timeout (xử lý phía máy chủ).
\end{itemize}

\subsection{UC03 -- Thêm tài khoản}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên use case}: Thêm tài khoản.
    \item \textbf{Mã}: UC03.
    \item \textbf{Tác nhân chính}: Quản trị viên.
    \item \textbf{Mô tả tóm tắt}: Quản trị viên tạo tài khoản mới cho nhân sự (QL kho, NV kho hoặc QTV khác), gán vai trò để phân quyền sử dụng hệ thống.
    \item \textbf{Mục tiêu}: Có thêm một tài khoản hợp lệ trong hệ thống, sẵn sàng đăng nhập.
    \item \textbf{Điều kiện trước}:
    \begin{enumerate}
        \item Quản trị viên đã đăng nhập (UC01) với vai trò QTV.
        \item Có thông tin nhân sự cần cấp tài khoản (họ tên, vai trò dự kiến).
    \end{enumerate}
    \item \textbf{Điều kiện sau}: Bản ghi \texttt{NguoiDung} mới được lưu; mật khẩu đã mã hóa; tài khoản có thể đăng nhập nếu trạng thái ``Hoạt động''.
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item QTV chọn menu ``Quản lý tài khoản''.
        \item Hệ thống hiển thị danh sách tài khoản hiện có.
        \item QTV chọn ``Thêm tài khoản''.
        \item Hệ thống mở form nhập: họ tên, tên đăng nhập, mật khẩu, nhập lại mật khẩu, email (nếu có), vai trò, trạng thái.
        \item QTV nhập đủ thông tin và nhấn ``Lưu''.
        \item Hệ thống kiểm tra các trường bắt buộc và định dạng.
        \item Hệ thống kiểm tra tên đăng nhập chưa tồn tại trong CSDL.
        \item Hệ thống mã hóa mật khẩu và lưu tài khoản mới.
        \item Hệ thống (tuỳ chọn) ghi log thao tác thêm tài khoản.
        \item Hệ thống thông báo thành công và cập nhật lại danh sách tài khoản.
    \end{enumerate}
    \item \textbf{Luồng thay thế}:
    \begin{enumerate}
        \item \textit{A1 -- Tên đăng nhập trùng}: Thông báo lỗi, yêu cầu đổi tên đăng nhập, giữ các trường khác.
        \item \textit{A2 -- Mật khẩu nhập lại không khớp}: Báo lỗi, không lưu.
        \item \textit{A3 -- Thiếu trường bắt buộc}: Đánh dấu trường thiếu, không gửi lưu.
    \end{enumerate}
    \item \textbf{Ghi chú}: Sửa tài khoản, khóa/mở khóa, tìm kiếm tài khoản có luồng tương tự (chọn bản ghi $\rightarrow$ thao tác $\rightarrow$ lưu) và không tách thành use case riêng trong phạm vi báo cáo.
\end{itemize}

\subsection{UC04 -- Thêm nhà cung cấp}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên use case}: Thêm nhà cung cấp.
    \item \textbf{Mã}: UC04.
    \item \textbf{Tác nhân chính}: Nhân viên kho, Quản lý kho.
    \item \textbf{Mô tả tóm tắt}: Bổ sung nhà cung cấp mới vào danh mục để dùng khi lập phiếu nhập.
    \item \textbf{Mục tiêu}: Lưu được bản ghi NCC mới, chọn được trên form nhập kho.
    \item \textbf{Điều kiện trước}: Actor đã đăng nhập với quyền quản lý danh mục đối tác.
    \item \textbf{Điều kiện sau}: Có bản ghi \texttt{NhaCungCap} mới; xuất hiện trong danh sách và form chọn.
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item Actor mở ``Quản lý nhà cung cấp'' (có thể từ UC08 xem danh sách).
        \item Chọn ``Thêm mới''.
        \item Nhập mã NCC, tên, địa chỉ, SĐT, email, người liên hệ, mô tả (nếu có).
        \item Nhấn ``Lưu''. Hệ thống kiểm tra mã bắt buộc và không trùng.
        \item Hệ thống lưu NCC, thông báo thành công, làm mới danh sách.
    \end{enumerate}
    \item \textbf{Luồng thay thế}:
    \begin{enumerate}
        \item \textit{A1 -- Mã trùng}: Báo lỗi, yêu cầu đổi mã.
        \item \textit{A2 -- Thiếu mã/tên}: Không cho lưu.
    \end{enumerate}
\end{itemize}

\subsection{UC05 -- Sửa nhà cung cấp}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên use case}: Sửa nhà cung cấp.
    \item \textbf{Mã}: UC05.
    \item \textbf{Tác nhân chính}: Nhân viên kho, Quản lý kho.

\subsection{UC13 -- Lập phiếu nhập kho}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên use case}: Lập phiếu nhập kho.
    \item \textbf{Mã}: UC13.
    \item \textbf{Tác nhân chính}: Nhân viên kho (lập và gửi duyệt).
    \item \textbf{Tác nhân phụ}: Quản lý kho (duyệt hoặc từ chối).
    \item \textbf{Mô tả tóm tắt}: Ghi nhận lô hàng nhập từ nhà cung cấp vào một kho cụ thể. Chỉ khi quản lý duyệt, tồn kho mới tăng -- tránh cập nhật tồn khi dữ liệu phiếu chưa được kiểm soát.
    \item \textbf{Mục tiêu}: Tạo phiếu nhập đầy đủ chi tiết; sau duyệt thì tồn kho phản ánh đúng số lượng đã nhập.
    \item \textbf{Điều kiện trước}:
    \begin{enumerate}
        \item NV và QL đã có tài khoản và đăng nhập.
        \item Đã có nhà cung cấp, kho và ít nhất một mặt hàng trong danh mục.
    \end{enumerate}
    \item \textbf{Điều kiện sau (duyệt thành công)}:
    \begin{enumerate}
        \item Phiếu ở trạng thái ``Đã duyệt''.
        \item Từng dòng hàng: \texttt{TonKho} tại kho nhập được cộng đúng số lượng.
        \item Có log thao tác duyệt.
    \end{enumerate}
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item NV chọn ``Lập phiếu nhập kho''.
        \item Hệ thống mở form phiếu: chọn nhà cung cấp, kho nhập, ngày nhập, ghi chú.
        \item NV thêm từng dòng chi tiết: chọn hàng hóa, nhập số lượng, đơn giá.
        \item Hệ thống tính thành tiền từng dòng và tổng tiền phiếu.
        \item NV kiểm tra lại và chọn ``Gửi duyệt''.
        \item Hệ thống kiểm tra phiếu có ít nhất một dòng hàng, các trường bắt buộc hợp lệ.
        \item Hệ thống lưu phiếu với trạng thái ``Chờ duyệt''.
        \item QL mở danh sách phiếu chờ duyệt, chọn phiếu cần xem.
        \item QL kiểm tra thông tin NCC, kho, danh sách hàng; chọn ``Duyệt''.
        \item Hệ thống đổi trạng thái phiếu thành ``Đã duyệt''.
        \item Với mỗi dòng chi tiết, hệ thống tăng \texttt{SoLuongTon} tương ứng (Kho, HangHoa).
        \item Hệ thống ghi log và thông báo duyệt thành công.
    \end{enumerate}
    \item \textbf{Luồng thay thế}:
    \begin{enumerate}
        \item \textit{A1 -- Phiếu không có dòng hàng}: Không cho gửi duyệt, yêu cầu bổ sung.
        \item \textit{A2 -- QL từ chối}: Nhập lý do từ chối; trạng thái ``Từ chối''; \textbf{không} đổi tồn kho.
        \item \textit{A3 -- NV hủy phiếu ở trạng thái cho phép}: Phiếu ``Đã hủy'', tồn không đổi.
    \end{enumerate}
\end{itemize}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên use case}: Lập phiếu kiểm kê.
    \item \textbf{Mã}: UC15.
    \item \textbf{Tác nhân chính}: Nhân viên kho (tạo phiếu, nhập số thực tế).
    \item \textbf{Tác nhân phụ}: Quản lý kho (duyệt điều chỉnh chênh lệch).
    \item \textbf{Mô tả tóm tắt}: Định kỳ đối chiếu số lượng hàng thực tế trong kho với số trên hệ thống. Chênh lệch chỉ được đưa vào tồn chính thức sau khi quản lý duyệt.
    \item \textbf{Mục tiêu}: Ghi nhận kết quả kiểm đếm; đồng bộ tồn kho với thực tế khi được duyệt.
    \item \textbf{Điều kiện trước}: Đã có kho và hàng; actor đã đăng nhập.
    \item \textbf{Điều kiện sau (duyệt)}: \texttt{TonKho} được gán bằng số thực tế; phiếu ``Đã duyệt''; có log.
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item NV chọn ``Lập phiếu kiểm kê'', chọn kho và ngày kiểm.
        \item Hệ thống nạp danh sách hàng đang theo dõi tại kho kèm \textbf{số lượng sổ} (từ \texttt{TonKho}).
        \item NV nhập \textbf{số lượng thực tế} sau khi kiểm đếm.
        \item Hệ thống tính \textbf{chênh lệch} = thực tế $-$ sổ sách cho từng dòng.
        \item NV gửi duyệt phiếu kiểm kê.
        \item QL xem tổng hợp chênh lệch (thừa/thiếu), chọn ``Duyệt điều chỉnh''.
        \item Hệ thống cập nhật tồn = số thực tế, đổi trạng thái phiếu, ghi log.
    \end{enumerate}
    \item \textbf{Luồng thay thế}: QL từ chối / yêu cầu kiểm lại $\rightarrow$ tồn giữ nguyên số sổ.
\end{itemize}

\subsection{UC16 -- Xem báo cáo nhập xuất tồn}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên use case}: Xem báo cáo nhập xuất tồn.
    \item \textbf{Mã}: UC16.
    \item \textbf{Tác nhân chính}: Quản lý kho, Quản trị viên.
    \item \textbf{Mô tả tóm tắt}: Tra cứu số liệu tổng hợp phục vụ điều hành kho: tồn hiện tại hoặc biến động nhập--xuất--tồn theo khoảng thời gian, theo kho hoặc theo hàng. Không bao gồm phân hệ tài chính/thanh toán phức tạp.
    \item \textbf{Mục tiêu}: Có bảng số liệu NXT/tồn đúng điều kiện lọc để xem hoặc xuất file.
    \item \textbf{Điều kiện trước}: Actor đã đăng nhập với quyền xem báo cáo; hệ thống đã có dữ liệu phiếu/tồn (có thể rỗng).
    \item \textbf{Điều kiện sau}: Actor xem được báo cáo; việc xuất Excel (nếu có) tạo file phía client; không bắt buộc ghi thay đổi nghiệp vụ.
    \item \textbf{Luồng sự kiện chính}:
    \begin{enumerate}
        \item Actor mở chức năng ``Báo cáo nhập xuất tồn''.
        \item Chọn loại xem: tồn hiện tại \textit{hoặc} nhập--xuất--tồn theo kỳ.
        \item Nhập điều kiện: từ ngày -- đến ngày, kho (tuỳ chọn), nhóm hàng / mã hàng (tuỳ chọn).
        \item Nhấn ``Xem báo cáo''.
        \item Hệ thống tổng hợp từ \texttt{PhieuNhapKho}/\texttt{ChiTietPhieuNhap}, \texttt{PhieuXuatKho}/\texttt{ChiTietPhieuXuat}, \texttt{TonKho}, \texttt{HangHoa}.
        \item Hệ thống hiển thị bảng kết quả (mã hàng, tên, tồn đầu/nhập/xuất/tồn cuối hoặc tồn hiện tại tùy loại).
        \item Actor có thể nhấn ``Xuất Excel'' để tải file.
    \end{enumerate}
    \item \textbf{Luồng thay thế}: Không có dữ liệu trong kỳ $\rightarrow$ bảng rỗng / thông báo không có phát sinh.
\end{itemize}

\section{Kết luận chương}

Chương 3 hoàn thành \textbf{phân tích chức năng}: dùng \textbf{biểu đồ use case} (tổng quát, phân rã) và \textbf{đặc tả use case} đầy đủ để xác định chức năng và mục tiêu của từng actor. Tiếp theo, Chương 4 thực hiện \textbf{phân tích cấu trúc} bằng \textbf{biểu đồ lớp}.
